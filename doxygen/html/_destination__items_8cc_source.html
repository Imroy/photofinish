100  void D_JPEG::add_variables(multihash& vars) {
101  if (!_quality.defined()) {
102  auto vi = vars.find("qual");
103  if (vi != vars.end()) {
104  _quality = boost::lexical_cast<int>(vi->second[0]);
105  vars.erase(vi);
106  set_defined();
107  }
108  }
109  if (!_sample.defined()) {
110  auto vi = vars.find("sample");
111  if (vi != vars.end()) {
112  std::string sample = vi->second[0];
113  size_t x = sample.find_first_of("x×");
114  if (x != std::string::npos) {
115  try {
116  int h = boost::lexical_cast<int>(sample.substr(0, x));
117  int v = boost::lexical_cast<int>(sample.substr(x + 1, sample.length() - x - 1));
118  _sample = std::pair<int, int>(h, v);
119  vars.erase(vi);
120  set_defined();
121  } catch (boost::bad_lexical_cast &ex) {
122  std::cerr << ex.what();
123  }
124  } else
125  std::cerr << "D_JPEG: Failed to parse sample \"" << vi->second[0] << "\"." << std::endl;
126  }
127  }
