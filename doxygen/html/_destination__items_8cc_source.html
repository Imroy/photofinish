114  void D_JPEG::add_variables(multihash& vars) {
115  if (!_quality.defined()) {
116  auto vi = vars.find("qual");
117  if (vi != vars.end()) {
118  _quality = boost::lexical_cast<int>(vi->second[0]);
119  vars.erase(vi);
120  set_defined();
121  }
122  }
123  if (!_sample.defined()) {
124  auto vi = vars.find("sample");
125  if (vi != vars.end()) {
126  std::string sample = vi->second[0];
127  size_t x = sample.find_first_of("x×");
128  if (x != std::string::npos) {
129  try {
130  int h = boost::lexical_cast<int>(sample.substr(0, x));
131  int v = boost::lexical_cast<int>(sample.substr(x + 1, sample.length() - x - 1));
132  _sample = std::pair<int, int>(h, v);
133  vars.erase(vi);
134  set_defined();
135  } catch (boost::bad_lexical_cast &ex) {
136  std::cerr << ex.what();
137  }
138  } else
139  std::cerr << "D_JPEG: Failed to parse sample \"" << vi->second[0] << "\"." << std::endl;
140  }
141  }
