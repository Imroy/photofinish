46 void make_preview(Image::ptr orig_image, Destination::ptr orig_dest, Tags::ptr filetags, ImageWriter::ptr preview_file, bool can_free = false) {
47  CMS::ColourModel orig_model = orig_image->format().colour_model();
48 
49  {
50  CMS::Format internal_format;
51  internal_format.set_colour_model(CMS::ColourModel::Lab);
52  SET_SAMPLE_FORMAT(internal_format);
53  orig_image->transform_colour_inplace(CMS::Profile::Lab4(), internal_format);
54  }
55 
56  auto resized_dest = orig_dest->dupe();
57 
58  resized_dest->set_depth(8);
59  resized_dest->clear_profile();
60  resized_dest->set_jpeg(D_JPEG(60, 2, 2, true));
61  resized_dest->jp2().set_rates({ 120, 12 });
62  resized_dest->jp2().set_tile_size(1024, 1024);
63  resized_dest->jp2().set_numresolutions(5);
64  resized_dest->tiff().set_compression("deflate");
65  resized_dest->webp().set_preset("photo");
66  resized_dest->webp().set_quality(80);
67  resized_dest->webp().set_method(6);
68 
69  auto frame = std::make_shared<Frame>(orig_image->width() * 0.25, orig_image->height() * 0.25,
70  0, 0,
71  orig_image->width(), orig_image->height());
72 
73  auto resized_image = frame->crop_resize(orig_image, D_resize::lanczos(3), true);
74 
75  CMS::Format resized_format = resized_image->format();
76  resized_format.set_colour_model(orig_model);
77  CMS::Format dest_format = preview_file->preferred_format(resized_dest->modify_format(resized_format));
78  CMS::Profile::ptr dest_profile = resized_dest->get_profile(dest_format.colour_model(), "preview");
79  resized_image->transform_colour_inplace(dest_profile, dest_format);
80 
81  filetags->copy_to(resized_image);
82  preview_file->write(resized_image, resized_dest, can_free);
83 }

85 void preview_dir(fs::path dir, std::string format, std::shared_ptr<Tags> tags) {
86  std::vector<fs::directory_entry> dir_list;
87  for (auto di = fs::directory_iterator(dir); di != fs::directory_iterator(); di++)
88  dir_list.push_back(*di);
89  sort(dir_list.begin(), dir_list.end());
90 
91  for (auto di : dir_list) {
92  std::cerr << di << std::endl;
93  try {
94  ImageFilepath infilepath(di.path());
95  ImageFilepath preview_filepath(di.path().filename(), format);
96 
97  if (exists(preview_filepath)
98  && (last_write_time(preview_filepath) > last_write_time(infilepath)))
99  continue;
100 
101  auto infile = ImageReader::open(infilepath);

103 
104  auto orig_dest = std::make_shared<Destination>();

106  auto filetags = tags->dupe();
107  filetags->copy_from(orig_image);
108
