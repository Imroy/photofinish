95  if (dest->jxr().defined() && dest->jxr().progressive().defined())

97 
98  if (img->format().colour_model() == CMS::ColourModel::Greyscale)
99  wmiSCP.cfColorFormat = Y_ONLY;
100  else {
101  wmiSCP.cfColorFormat = YUV_444;
102  if (dest->jxr().defined()) {
103  std::string sub = dest->jxr().subsampling();
104  if (boost::iequals(sub, "420"))
105  wmiSCP.cfColorFormat = YUV_420;
106  else if (boost::iequals(sub, "422"))
107  wmiSCP.cfColorFormat = YUV_422;
108  else if (boost::iequals(sub, "CMYK"))
109  wmiSCP.cfColorFormat = CMYK;
110  }
111  }
112 
113  float iq_float = (dest->jxr().defined() ? dest->jxr().quality() : 100) / 100.0f;
114 
115  if (iq_float == 1.0f)
116  wmiSCP.olOverlap = OL_NONE;
117  else {
118  wmiSCP.olOverlap = iq_float > 0.4f ? OL_ONE : OL_TWO;
119  if (dest->jxr().defined() && dest->jxr().overlap().defined()) {
120  std::string ol = dest->jxr().overlap();
121  if (boost::iequals(ol, "none"))
122  wmiSCP.olOverlap = OL_NONE;
123  else if (boost::iequals(ol, "one") || (ol == "1"))
124  wmiSCP.olOverlap = OL_ONE;
125  else if (boost::iequals(ol, "two") || (ol == "2"))
126  wmiSCP.olOverlap = OL_TWO;
127  }
128  }
129 
130  if (iq_float == 1.0f)
131  wmiSCP.uiDefaultQPIndex = 1;
132  else {
133  float iq;
134  if (iq_float > 0.8f)
135  iq = 0.8f + (iq_float - 0.8f) * 1.5f;
136  else
137  iq = iq_float;
138 
139  int qi = 10.0f * iq;
140  float qf = 10.0f * iq - (float)qi;
141 
142  int *qp_row = jxr_qp_table[qi];
143 
144  wmiSCP.uiDefaultQPIndex = (U8)(0.5f + qp_row[0] * (1.0f - qf) + (qp_row + 6)[0] * qf);
145  wmiSCP.uiDefaultQPIndexU = (U8)(0.5f + qp_row[1] * (1.0f - qf) + (qp_row + 6)[1] * qf);
146  wmiSCP.uiDefaultQPIndexV = (U8)(0.5f + qp_row[2] * (1.0f - qf) + (qp_row + 6)[2] * qf);
147  wmiSCP.uiDefaultQPIndexYHP = (U8)(0.5f + qp_row[3] * (1.0f - qf) + (qp_row + 6)[3] * qf);
148  wmiSCP.uiDefaultQPIndexUHP = (U8)(0.5f + qp_row[4] * (1.0f - qf) + (qp_row + 6)[4] * qf);
149  wmiSCP.uiDefaultQPIndexVHP = (U8)(0.5f + qp_row[5] * (1.0f - qf) + (qp_row + 6)[5] * qf);
150  }
151 
152  wmiSCP.cNumOfSliceMinus1H = wmiSCP.cNumOfSliceMinus1V = 0;
153  if (dest->jxr().defined() && dest->jxr().tilesize().defined()) {
154  int tile_size = dest->jxr().tilesize();
155 
156  // Strip all but the MSB
157  for (int i = 1; i < tile_size; i <<= 1)
158  tile_size &= (i ^ 0x7fffffffL);
159 
160  if (tile_size >= 256) {
161  if (tile_size > 1024)
162  tile_size = 1024;
163  std::cerr << "\tTile size = " << tile_size << std::endl;
164 
165  int i = 0;
166  unsigned int p = 0;
167  for (i = 0; i < MAX_TILES; i++) {
168  wmiSCP.uiTileY[i] = tile_size << 4;
169  p += tile_size;
170  if (p >= img->height())
171  break;
172  }
173 
174  wmiSCP.cNumOfSliceMinus1H = i;
175 
176  p = 0;
177  for (i = 0; i < MAX_TILES; i++) {
178  wmiSCP.uiTileX[i] = tile_size << 4;
179  p += tile_size;
180  if (p >= img->width())
181  break;
182  }
183 
184  wmiSCP.cNumOfSliceMinus1V = i;
185  }
186  }

190  PKPixelFormatGUID pixel_format = jxr_pixel_format(img->format());
191  JXRcheck(encoder->SetPixelFormat(encoder, pixel_format));
192 
193  JXRcheck(encoder->SetSize(encoder, img->width(), img->height()));
194  JXRcheck(encoder->SetResolution(encoder, img->xres(), img->yres()));
195

201  }
