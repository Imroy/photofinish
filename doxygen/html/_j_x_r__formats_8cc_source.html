51  CMS::Format jxr_cms_format(const PKPixelFormatGUID &g) {
52  long rc;
53 
54  PKPixelInfo pi;
55  pi.pGUIDPixFmt = &g;
56  JXRcheck(PixelFormatLookup(&pi, LOOKUP_FORWARD));
57  CMS::Format f;
58  switch (pi.cfColorFormat) {
59  case Y_ONLY:
60  f.set_colour_model(CMS::ColourModel::Greyscale);
61  break;
62 
63  case YUV_420:
64  case YUV_422:
65  case YUV_444:
66  case CF_RGB:
67  f.set_colour_model(CMS::ColourModel::RGB);
68  break;
69 
70  case CMYK:
71  f.set_colour_model(CMS::ColourModel::CMYK);
72  break;
73 
74  default:
75  std::cerr << "Unhandled JPEG XR colour format " << (int)pi.cfColorFormat << std::endl;
76  break;
77  }
78 
79  switch (pi.bdBitDepth) {
80  case BD_8:
81  f.set_8bit();
82  break;
83 
84  case BD_16:
85  f.set_16bit();
86  break;
87 
88  case BD_32:
89  f.set_32bit();
90  break;
91 
92  case BD_16F:
93  f.set_half();
94  break;
95 
96  case BD_32F:
97  f.set_float();
98  break;
99 
100  default:
101  std::cerr << "Unhandled JPEG XR bit depth " << (int)pi.bdBitDepth << std::endl;
102  }
103 
104  if (pi.grBit & PK_pixfmtHasAlpha)
105  f.set_extra_channels(1);
106 
107  if (pi.grBit & PK_pixfmtPreMul)
108  f.set_premult_alpha();
109 
110  if (pi.grBit & PK_pixfmtBGR)
111  f.set_swap();
112 
113  return f;
114  }
