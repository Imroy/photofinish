59  cmsHPROFILE Image::default_profile(cmsUInt32Number default_type) {
60  cmsHPROFILE profile = NULL;
61  switch (T_COLORSPACE(default_type)) {
62  case PT_RGB:
63  std::cerr << "\tUsing default sRGB profile." << std::endl;
64  profile = cmsCreate_sRGBProfile();
65  break;
66 
67  case PT_GRAY:
68  std::cerr << "\tUsing default greyscale profile." << std::endl;
69  // Build a greyscale profile with the same gamma curve and white point as sRGB.
70  // Copied from cmsCreate_sRGBProfileTHR() and Build_sRGBGamma().
71  {
72  cmsCIExyY D65;
73  cmsWhitePointFromTemp(&D65, 6504);
74  double Parameters[5] = {
75  2.4, // Gamma
76  1.0 / 1.055, // a
77  0.055 / 1.055, // b
78  1.0 / 12.92, // c
79  0.04045, // d
80  };
81  // y = (x >= d ? (a*x + b)^Gamma : c*x)
82  cmsToneCurve *gamma = cmsBuildParametricToneCurve(NULL, 4, Parameters);
83  profile = cmsCreateGrayProfile(&D65, gamma);
84  cmsFreeToneCurve(gamma);
85 
86  cmsMLU *DescriptionMLU = cmsMLUalloc(NULL, 1);
87  if (DescriptionMLU != NULL) {
88  if (cmsMLUsetWide(DescriptionMLU, "en", "AU", L"sGrey built-in"))
89  cmsWriteTag(profile, cmsSigProfileDescriptionTag, DescriptionMLU);
90  cmsMLUfree(DescriptionMLU);
91  }
92  }
93  break;
94 
95  case PT_Lab:
96  std::cerr << "\tUsing default Lab profile." << std::endl;
97  profile = cmsCreateLab4Profile(NULL);
98  break;
99 
100  default:
101  std::cerr << "** Cannot assign a default profile for colour space " << T_COLORSPACE(default_type) << " **" << std::endl;
102  }
103 
104  return profile;
105  }

107  Image::ptr Image::transform_colour(cmsHPROFILE dest_profile, cmsUInt32Number dest_type, cmsUInt32Number intent, bool can_free) {
108  std::cerr << "Transforming colour..." << std::endl;
109 
110  cmsHPROFILE profile = _profile;
111  if (_profile == NULL)
112  profile = default_profile(_type);
113 
114  cmsHTRANSFORM transform = cmsCreateTransform(profile, _type,
115  dest_profile, dest_type,
116  intent, 0);
117  if (_profile == NULL)
118  cmsCloseProfile(profile);
119 
120  auto dest = std::make_shared<Image>(_width, _height, dest_type);
121  dest->set_profile(dest_profile);
122  dest->set_resolution(_xres, _yres);
123 
124 #pragma omp parallel for schedule(dynamic, 1)
125  for (unsigned int y = 0; y < _height; y++) {
126  cmsDoTransform(transform, _rowdata[y], dest->row(y), _width);
127  if (can_free)
128  this->free_row(y);
129  }
130 
131  cmsDeleteTransform(transform);
132 
133  return dest;
134  }

136  void Image::transform_colour_inplace(cmsHPROFILE dest_profile, cmsUInt32Number dest_type, cmsUInt32Number intent) {
137 #pragma omp parallel
138  {
139 #pragma omp master

141  std::cerr << "Transforming colour in-place using " << omp_get_num_threads() << " threads..." << std::endl;
142  }
143  }
144 
145  cmsHPROFILE profile = _profile;
146  if (_profile == NULL)
147  profile = default_profile(_type);
148 
149  cmsHTRANSFORM transform = cmsCreateTransform(profile, _type,
150  dest_profile, dest_type,
151  intent, 0);
152  if (_profile == NULL)
153  cmsCloseProfile(profile);
154 
155  unsigned char dest_bytes = T_BYTES(dest_type);
156  if (T_FLOAT(dest_type) && (dest_bytes == 0))
157  dest_bytes = 8;
158  size_t dest_pixel_size = (T_CHANNELS(dest_type) + T_EXTRA(dest_type)) * dest_bytes;
159  size_t dest_row_size = _width * dest_pixel_size;
160 
161 #pragma omp parallel for schedule(dynamic, 1)
162  for (unsigned int y = 0; y < _height; y++) {
163  _check_rowdata_alloc(y);
164  unsigned char *src_rowdata = _rowdata[y];
165  unsigned char *dest_rowdata = (unsigned char*)malloc(dest_row_size);
166  cmsDoTransform(transform, src_rowdata, dest_rowdata, _width);
167  _rowdata[y] = dest_rowdata;
168  free(src_rowdata);
169 
170  if (omp_get_thread_num() == 0)
171  std::cerr << "\r\tTransformed " << y + 1 << " of " << _height << " rows";
172  }
173  std::cerr << "\r\tTransformed " << _height << " of " << _height << " rows." << std::endl;
174 
175  cmsDeleteTransform(transform);
176 
177  _profile = dest_profile;
178  _type = dest_type;
179  _pixel_size = dest_pixel_size;
180  _row_size = dest_row_size;
181  }
