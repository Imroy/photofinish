82  void transfer_alpha_typed2(unsigned int width, unsigned char src_channels, const A* src_row, unsigned char dest_channels, const B* dest_row) {
83  double factor = (double)scaleval<B>() / scaleval<A>();
84  A *inp = const_cast<A*>(src_row) + src_channels;
85  B *outp = const_cast<B*>(dest_row) + dest_channels;
86  for (unsigned int x = width; x; x--, inp += 1 + src_channels, outp += 1 + dest_channels)
87  *outp = limitval<B>((*inp) * factor);
88  }

281

333  template <typename SRC, typename DST>
334  void Image::_alpha_mult_src_dst(CMS::Format dest_format) {
335 #pragma omp parallel
336  {
337 #pragma omp master
338  {
339  std::cerr << "Multiplying colour from the alpha channel and transforming into " << dest_format << " (scale=" << (SAMPLE)scaleval<DST>() << "/" << (SAMPLE)scaleval<SRC>() << ") using " << omp_get_num_threads() << " threads..." << std::endl;
340  }

342

344  size_t dest_row_size = _width * dest_pixel_size;
345  unsigned char alphachan = _format.channels();
346  SAMPLE scale = (SAMPLE)scaleval<DST>() / scaleval<SRC>();
347  SAMPLE src_scale = scale / scaleval<SRC>();
348 
349 #pragma omp parallel for schedule(dynamic, 1)
350  for (unsigned int y = 0; y < _height; y++) {
351  void *src_rowdata = row(y);
352  void *dest_rowdata = malloc(dest_row_size);

356  for (unsigned int x = 0; x < _width; x++, in += _format.total_channels(), out += dest_format.total_channels()) {
357  SAMPLE alpha = in[alphachan] * src_scale;
358  unsigned char c;
359  for (c = 0; c < alphachan; c++) {
360  out[c] = limitval<DST>(in[c] * alpha);
361  }
362  for (; c < dest_format.total_channels(); c++)
363  out[c] = limitval<DST>(in[c] * scale);
364  }
365 
366  _rowdata[y] = (unsigned char*)dest_rowdata;
367  free(src_rowdata);
368 
369  if (omp_get_thread_num() == 0)
370  std::cerr << "\r\tTransformed " << y + 1 << " of " << _height << " rows";
371  }
372  std::cerr << "\r\tTransformed " << _height << " of " << _height << " rows." << std::endl;
373 
374  _format.set_channel_type(dest_format);
375  _format.set_premult_alpha();
376  _format.set_packed();
377  _pixel_size = dest_pixel_size;
378  _row_size = dest_row_size;
