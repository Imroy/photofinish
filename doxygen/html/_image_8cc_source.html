82  template <typename A, typename B>
83  void transfer_alpha_typed2(unsigned int width, unsigned char src_channels, const A* src_row, unsigned char dest_channels, const B* dest_row) {
84  double factor = (double)scaleval<B>() / scaleval<A>();
85  A *inp = const_cast<A*>(src_row) + src_channels;
86  B *outp = const_cast<B*>(dest_row) + dest_channels;
87  for (unsigned int x = width; x; x--, inp += 1 + src_channels, outp += 1 + dest_channels)
88  *outp = limitval<B>((*inp) * factor);
89  }

91  template <typename A>
92  void transfer_alpha_typed(unsigned int width, unsigned char src_channels, const A* src_row, CMS::Format dest_format, const unsigned char* dest_row) {
93  unsigned char dest_channels = (unsigned char)dest_format.channels();
94  switch (dest_format.bytes_per_channel()) {
95  case 1:
96  transfer_alpha_typed2<A, unsigned char>(width, src_channels, src_row, dest_channels, (unsigned char*)dest_row);
97  break;
98 
99  case 2:
100  transfer_alpha_typed2<A, short unsigned int>(width, src_channels, src_row, dest_channels,(short unsigned int*)dest_row);
101  break;
102 
103  case 4:
104  if (dest_format.is_fp())
105  transfer_alpha_typed2<A, float>(width, src_channels, src_row, dest_channels,(float*)dest_row);
106  else
107  transfer_alpha_typed2<A, unsigned int>(width, src_channels, src_row, dest_channels,(unsigned int*)dest_row);
108  break;
109 
110  case 8:
111  transfer_alpha_typed2<A, double>(width, src_channels, src_row, dest_channels,(double*)dest_row);
112  break;
113 
114  }

284  template <typename SRC>
285  void Image::_un_alpha_mult_src(void) {
286  CMS::Format dest_format = _format;
287  SET_SAMPLE_FORMAT(dest_format);
288 
289 #pragma omp parallel
290  {
291 #pragma omp master
292  {
293  std::cerr << "Un-multiplying colour from the alpha channel and transforming into " << dest_format << " (scale=" << scaleval<SAMPLE>() << "/" << (SAMPLE)scaleval<SRC>() << ") using " << omp_get_num_threads() << " threads..." << std::endl;
294  }
295  }
296 
297  size_t dest_pixel_size = dest_format.bytes_per_pixel();
298  size_t dest_row_size = _width * dest_pixel_size;
299  unsigned char alphachan = _format.channels();
300  SAMPLE scale = scaleval<SAMPLE>() / scaleval<SRC>();
301 
302  Timer timer;
303  timer.start();
304 
305 #pragma omp parallel for schedule(dynamic, 1)
306  for (unsigned int y = 0; y < _height; y++) {
307  unsigned char *src_rowdata = row(y);
308  unsigned char *dest_rowdata = new unsigned char [dest_row_size];
309 
310  SRC *in = (SRC*)src_rowdata;
311  SAMPLE *out = (SAMPLE*)dest_rowdata;
312  for (unsigned int x = 0; x < _width; x++, in += _format.total_channels(), out += dest_format.total_channels()) {
313  SRC alpha = in[alphachan];
314  unsigned char c;
315  if (alpha > 0) {
316  SAMPLE recip_alpha = scaleval<SAMPLE>() / alpha;
317  for (c = 0; c < alphachan; c++)
318  out[c] = limitval<SAMPLE>(in[c] * recip_alpha);
319  } else
320  for (c = 0; c < alphachan; c++)
321  out[c] = 0; // or scaleval<SAMPLE>() ?
322  for (; c < _format.total_channels(); c++)
323  out[c] = limitval<SAMPLE>(in[c] * scale);
324  }
325 
326  _rowdata[y] = dest_rowdata;
327  delete [] src_rowdata;
328 
329  if (omp_get_thread_num() == 0)
330  std::cerr << "\r\tTransformed " << y + 1 << " of " << _height << " rows";
331  }
332  timer.stop();
333  std::cerr << "\r\tTransformed " << _height << " of " << _height << " rows." << std::endl;
334 
335  if (benchmark_mode) {
336  std::cerr << std::setprecision(2) << std::fixed;
337  long long pixel_count = _width * _height;
338  std::cerr << "Benchmark: Un-multiplied alpha of " << pixel_count << " pixels in " << timer << " = " << (pixel_count / timer.elapsed() / 1e+6) << " Mpixels/second" << std::endl;

340 
341  _format = dest_format;
342  _format.unset_premult_alpha();

363  template <typename SRC, typename DST>
364  void Image::_alpha_mult_src_dst(CMS::Format dest_format) {
365 #pragma omp parallel
366  {
367 #pragma omp master
368  {
369  std::cerr << "Multiplying colour from the alpha channel and transforming into " << dest_format << " (scale=" << (SAMPLE)scaleval<DST>() << "/" << (SAMPLE)scaleval<SRC>() << ") using " << omp_get_num_threads() << " threads..." << std::endl;
370  }
371  }
372 
373  size_t dest_pixel_size = dest_format.bytes_per_pixel();
374  size_t dest_row_size = _width * dest_pixel_size;
375  unsigned char alphachan = _format.channels();
376  SAMPLE scale = (SAMPLE)scaleval<DST>() / scaleval<SRC>();
377  SAMPLE src_scale = scale / scaleval<SRC>();
378 
379  Timer timer;
380  timer.start();
381 
382 #pragma omp parallel for schedule(dynamic, 1)
383  for (unsigned int y = 0; y < _height; y++) {
384  unsigned char *src_rowdata = row(y);

387  SRC *in = (SRC*)src_rowdata;
388  DST *out = (DST*)dest_rowdata;
389  for (unsigned int x = 0; x < _width; x++, in += _format.total_channels(), out += dest_format.total_channels()) {
390  SAMPLE alpha = in[alphachan] * src_scale;
391  unsigned char c;
392  for (c = 0; c < alphachan; c++) {
393  out[c] = limitval<DST>(in[c] * alpha);
394  }
395  for (; c < dest_format.total_channels(); c++)
396  out[c] = limitval<DST>(in[c] * scale);
397  }
398 
399  _rowdata[y] = dest_rowdata;
