100  template <typename Num_type, typename R_type>
101  Exiv2::ValueType<R_type>& closest_Rational(double value) {
102  double margin = fabs(value) * 1e-6;
103  Num_type num = 0;
104  Num_type den;
105  for (den = 1; den < INT_MAX; den++) {
106  double numf = value * den;
107  if ((numf < INT_MIN) || (numf > INT_MAX))
108  break;
109 
110  num = round(numf);
111  double error = fabs(num - numf);
112  if (error < margin * den)
113  break;
114  }
115 
116  Exiv2::ValueType<R_type> *rv = new Exiv2::ValueType<R_type>(R_type(num, den));
117  return *rv;
118  }
