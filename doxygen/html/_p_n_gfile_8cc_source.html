42  void png_info_cb(png_structp png, png_infop info) {
43  png_set_gamma(png, 1.0, 1.0);
44  png_set_alpha_mode(png, PNG_ALPHA_PNG, 1.0);
45  png_set_packing(png);
46  png_set_swap(png);
47  png_read_update_info(png, info);
48 
49  transform_queue *queue = (transform_queue*)png_get_progressive_ptr(png);
50 
51  unsigned int width, height;
52  int bit_depth, colour_type;
53  png_get_IHDR(png, info, &width, &height, &bit_depth, &colour_type, NULL, NULL, NULL);
54  std::cerr << "\t" << width << "×" << height << ", " << bit_depth << " bpp, type " << colour_type << "." << std::endl;
55 
56  Image::ptr img(new Image(width, height));
57  queue->destination()->set_depth(bit_depth);
58 
59  cmsUInt32Number cmsType;
60  switch (colour_type) {
61  case PNG_COLOR_TYPE_GRAY:
62  cmsType = COLORSPACE_SH(PT_GRAY) | CHANNELS_SH(1) | BYTES_SH(bit_depth >> 3);
63  img->set_greyscale();
64  break;
65  case PNG_COLOR_TYPE_RGB:
66  cmsType = COLORSPACE_SH(PT_RGB) | CHANNELS_SH(3) | BYTES_SH(bit_depth >> 3);
67  break;
68  default:
69  std::cerr << "** unsupported PNG colour type " << colour_type << " **" << std::endl;
70  exit(1);
71  }
72  queue->set_image(img, T_CHANNELS(cmsType));
73 
74  {
75  unsigned int xres, yres;
76  int unit_type;
77  if (png_get_pHYs(png, info, &xres, &yres, &unit_type)) {
78  switch (unit_type) {
79  case PNG_RESOLUTION_METER:
80  img->set_resolution(xres * 0.0254, yres * 0.0254);
81  break;
82  case PNG_RESOLUTION_UNKNOWN:
83  break;
84  default:
85  std::cerr << "** unknown unit type " << unit_type << " **" << std::endl;
86  }
87  if (img->xres().defined() && img->yres().defined())
88  std::cerr << "\tImage has resolution of " << img->xres() << "×" << img->yres() << " PPI." << std::endl;
89  }
90  }
91 
92  cmsHPROFILE lab = cmsCreateLab4Profile(NULL);
93  cmsHPROFILE profile = NULL;
94 
95  if (png_get_valid(png, info, PNG_INFO_iCCP)) {
96  std::cerr << "\tImage has iCCP chunk." << std::endl;
97  char *profile_name;
98  int compression_type;
99  png_bytep profile_data;
100  unsigned int profile_len;
101  if (png_get_iCCP(png, info, &profile_name, &compression_type, &profile_data, &profile_len) == PNG_INFO_iCCP) {
102  std::cerr << "\tLoading ICC profile \"" << profile_name << "\" from file..." << std::endl;
103  profile = cmsOpenProfileFromMem(profile_data, profile_len);
104  void *data_copy = malloc(profile_len);
105  memcpy(data_copy, profile_data, profile_len);
106  queue->destination()->set_profile(profile_name, data_copy, profile_len);
107  }
108  }
109  if (profile == NULL)
110  profile = ImageFile::default_profile(cmsType);
111 
112  cmsHTRANSFORM transform = cmsCreateTransform(profile, cmsType,
113  lab, IMAGE_TYPE,
114  INTENT_PERCEPTUAL, 0);
115  cmsCloseProfile(lab);
116  cmsCloseProfile(profile);
117 
118  queue->set_transform(transform);
119  }

143  Image::ptr PNGfile::read(Destination::ptr dest) const {
144  std::cerr << "Opening file " << _filepath << "..." << std::endl;
145  fs::ifstream fb(_filepath, std::ios_base::in);
146  if (fb.fail())
147  throw FileOpenError(_filepath.native());
148 
149  {
150  unsigned char header[8];
151  fb.read((char*)header, 8);
152  if (png_sig_cmp(header, 0, 8))
153  throw FileContentError(_filepath.string(), "is not a PNG file");
154  fb.seekg(0, std::ios_base::beg);
155  }
156 
157  png_structp png = png_create_read_struct(PNG_LIBPNG_VER_STRING,
158  NULL, NULL, NULL);
159  if (!png)
160  throw LibraryError("libpng", "Could not create PNG read structure");
161 
162  png_infop info = png_create_info_struct(png);
163  if (!info) {
164  png_destroy_read_struct(&png, (png_infopp)NULL, (png_infopp)NULL);
165  throw LibraryError("libpng", "Could not create PNG info structure");
166  }
167 
168  if (setjmp(png_jmpbuf(png))) {
169  png_destroy_read_struct(&png, &info, NULL);
170  fb.close();
171  throw LibraryError("libpng", "Something went wrong reading the PNG");
172  }
173 
174  transform_queue queue(dest);
175 
176 #pragma omp parallel shared(queue)
177  {
178  int th_id = omp_get_thread_num();
179  if (th_id == 0) { // Master thread
180  std::cerr << "\tReading PNG image and transforming into L*a*b* using " << omp_get_num_threads() << " threads..." << std::endl;
181  png_set_progressive_read_fn(png, (void *)&queue, png_info_cb, png_row_cb, png_end_cb);
182  png_byte buffer[1048576];
183  size_t length;
184  do {
185  fb.read((char*)buffer, 1048576);
186  length = fb.gcount();
187  png_process_data(png, info, buffer, length);
188  while (queue.num_rows() > 100)
189  queue.reader_process_row();
190  } while (length > 0);
191  std::cerr << std::endl;
192  queue.set_finished();
193  png_run_workqueue(&queue); // Help finish off the transforming of image data
194  } else {
195  png_run_workqueue(&queue); // Worker threads just run the workqueue
196  }
197  }
198  queue.free_transform();
199 
200  png_destroy_read_struct(&png, &info, NULL);
201  fb.close();
202 
203  std::cerr << "Done." << std::endl;
204  return queue.image();
205  }

219  void PNGfile::write(Image::ptr img, Destination::ptr dest, bool can_free) const {
220  std::cerr << "Opening file " << _filepath << "..." << std::endl;
221  fs::ofstream fb;
222  fb.open(_filepath, std::ios_base::out);
223 
224  png_structp png = png_create_write_struct(PNG_LIBPNG_VER_STRING,
225  NULL, NULL, NULL);
226  if (!png)
227  throw LibraryError("libpng", "Could not create PNG write structure");
228 
229  png_infop info = png_create_info_struct(png);
230  if (!info) {
231  png_destroy_write_struct(&png, (png_infopp)NULL);
232  throw LibraryError("libpng", "Could not create PNG info structure");
233  }
234 
235  if (setjmp(png_jmpbuf(png))) {
236  png_destroy_write_struct(&png, &info);
237  fb.close();
238  throw LibraryError("libpng", "Something went wrong writing the PNG");
239  }
240 
241  png_set_write_fn(png, &fb, png_write_ostream_cb, png_flush_ostream_cb);
242 
243  int png_colour_type, png_channels;
244  cmsUInt32Number cmsTempType;
245  if (img->is_colour()) {
246  png_colour_type = PNG_COLOR_TYPE_RGB;
247  png_channels = 3;
248  cmsTempType = COLORSPACE_SH(PT_RGB) | CHANNELS_SH(3) | BYTES_SH(2);
249  } else {
250  png_colour_type = PNG_COLOR_TYPE_GRAY;
251  png_channels = 1;
252  cmsTempType = COLORSPACE_SH(PT_GRAY) | CHANNELS_SH(1) | BYTES_SH(2);
253  }
254 
255  int depth = 8; // Default value
256  if (dest->depth().defined())
257  depth = dest->depth();
258 
259  std::cerr << "\tWriting header for " << img->width() << "×" << img->height()
260  << " " << depth << "-bit " << (png_channels == 1 ? "greyscale" : "RGB")
261  << " PNG image..." << std::endl;
262  png_set_IHDR(png, info,
263  img->width(), img->height(), depth, png_colour_type,
264  PNG_INTERLACE_NONE, PNG_COMPRESSION_TYPE_DEFAULT, PNG_FILTER_TYPE_DEFAULT);
265  png_set_filter(png, 0, PNG_ALL_FILTERS);
266  png_set_compression_level(png, Z_BEST_COMPRESSION);
267 
268  if (img->xres().defined() && img->yres().defined()) {
269  unsigned int xres = round(img->xres() / 0.0254);
270  unsigned int yres = round(img->yres() / 0.0254);
271  png_set_pHYs(png, info, xres, yres, PNG_RESOLUTION_METER);
272  }
273 
274  cmsUInt32Number intent = INTENT_PERCEPTUAL; // Default value
275  if (dest->intent().defined())
276  intent = dest->intent();
277 
278  cmsHPROFILE profile = NULL;
279  if (dest->profile()) {
280  profile = dest->profile()->profile();
281  if (dest->profile()->has_data()) {
282  std::cerr << "\tEmbedding profile \"" << dest->profile()->name().get() << "\" from data (" << dest->profile()->data_size() << " bytes)." << std::endl;
283  png_set_iCCP(png, info, dest->profile()->name()->c_str(), 0, (unsigned char*)dest->profile()->data(), dest->profile()->data_size());
284  }
285  } else {
286  if ((profile == NULL) && (img->is_greyscale())) {
287  std::cerr << "\tUsing default greyscale profile." << std::endl;
288  cmsToneCurve *gamma = cmsBuildGamma(NULL, 2.2);
289  profile = cmsCreateGrayProfile(cmsD50_xyY(), gamma);
290  cmsFreeToneCurve(gamma);
291  }
292 
293  if (profile != NULL) {
294  cmsUInt32Number len;
295  cmsSaveProfileToMem(profile, NULL, &len);
296  if (len > 0) {
297  png_bytep profile_data = (png_bytep)malloc(len);
298  if (cmsSaveProfileToMem(profile, profile_data, &len)) {
299  std::string profile_name = "icc"; // Default value
300  if (dest->profile() && dest->profile()->name().defined())
301  profile_name = dest->profile()->name();
302  std::cerr << "\tEmbedding profile \"" << profile_name << "\" (" << len << " bytes)." << std::endl;
303  png_set_iCCP(png, info, profile_name.c_str(), 0, profile_data, len);
304  }
305  }
306  } else {
307  std::cerr << "\tUsing default sRGB profile." << std::endl;
308  profile = cmsCreate_sRGBProfile();
309  png_set_sRGB_gAMA_and_cHRM(png, info, intent);
310  }
311  }
312 
313  {
314  time_t t = time(NULL);
315  if (t > 0) {
316  png_time ptime;
317  png_convert_from_time_t(&ptime, t);
318  std::cerr << "\tAdding time chunk." << std::endl;
319  png_set_tIME(png, info, &ptime);
320  }
321  }
322 
323  png_bytepp png_rows = (png_bytepp)malloc(img->height() * sizeof(png_bytep));
324  for (unsigned int y = 0; y < img->height(); y++)
325  png_rows[y] = (png_bytep)malloc(img->width() * png_channels * (depth >> 3));
326 
327  png_set_rows(png, info, png_rows);
328 
329  cmsHPROFILE lab = cmsCreateLab4Profile(NULL);
330  cmsHTRANSFORM transform = cmsCreateTransform(lab, IMAGE_TYPE,
331  profile, cmsTempType,
332  intent, 0);
333  cmsCloseProfile(lab);
334  cmsCloseProfile(profile);
335 
336  transform_queue queue(dest, img, png_channels, transform);
337  if (depth == 8)
338  for (unsigned int y = 0; y < img->height(); y++)
339  queue.add(y);
340  else
341  for (unsigned int y = 0; y < img->height(); y++)
342  queue.add(y, png_rows[y]);
343 
344 #pragma omp parallel shared(queue)
345  {
346  int th_id = omp_get_thread_num();
347  if (th_id == 0) { // Master thread
348  std::cerr << "\tTransforming image data from L*a*b* using " << omp_get_num_threads() << " threads." << std::endl;
349 
350  Ditherer ditherer(img->width(), png_channels);
351 
352  for (unsigned int y = 0; y < img->height(); y++) {
353  // Process rows until the one we need becomes available, or the queue is empty
354  {
355  short unsigned int *row = queue.row(y);
356  while (!queue.empty() && (row == NULL)) {
357  queue.writer_process_row();
358  row = queue.row(y);
359  }
360 
361  // If it's still not available, something has gone wrong
362  if (row == NULL) {
363  std::cerr << "** Oh crap (y=" << y << ", num_rows=" << queue.num_rows() << " **" << std::endl;
364  exit(2);
365  }
366 
367  if (can_free)
368  img->free_row(y);
369  if (depth == 8) {
370  ditherer.dither(row, png_rows[y], y == img->height() - 1);
371  queue.free_row(y);
372  }
373  }
374  std::cerr << "\r\tTransformed " << y + 1 << " of " << img->height() << " rows ("
375  << queue.num_rows() << " left) ";
376  }
377  std::cerr << std::endl;
378  } else { // Other thread(s) transform the image data
379  while (!queue.empty())
380  queue.writer_process_row();
381  }
382  }
383  cmsDeleteTransform(transform);
384 
385  std::cerr << "\tWriting PNG image data..." << std::endl;
386  png_write_png(png, info, PNG_TRANSFORM_SWAP_ENDIAN, NULL);
387 
388  for (unsigned int y = 0; y < img->height(); y++)
389  free(png_rows[y]);
390  free(png_rows);
391 
392  png_destroy_write_struct(&png, &info);
393  fb.close();
394 
395  std::cerr << "Done." << std::endl;
396  }
