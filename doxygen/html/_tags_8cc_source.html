68  void Tags::load(fs::path filepath) {
69  std::cerr << "Loading \"" << filepath.native() << "\"..." << std::endl;
70  std::ifstream fin(filepath.native());
71  if (fin.fail())
72  throw FileOpenError(filepath.native());
73 
74  while (!fin.eof()) {
75  std::string line;
76  getline(fin, line);
77  if (fin.eof())
78  break;
79 
80  // Remove comment text
81  {
82  size_t x;
83  if ((x = line.find_first_of('#', 1)) != std::string::npos)
84  line = line.substr(0, x - 1);
85  }
86 
87  if (line.substr(0, 2) == "#@") {
88  int start = line.find_first_not_of(" \t", 2);
89  int end = line.find_last_not_of(" \t");
90  fs::path filepath = line.substr(start, end + 1 - start);
91  if (!try_load(filepath))
92  std::cerr << "** Could not find a tag file named \"" << filepath << "\" in the search paths **" << std::endl;
93  continue;
94  }
95 
96  if (line.substr(0, 2) == "#$") {
97  int start = line.find_first_not_of(" \t", 2);
98  int eq = line.find_first_of('=', start);
99  int end = line.find_last_not_of(" \t");
100  std::string key = line.substr(start, eq - start);
101  std::string value = line.substr(eq + 1, end - eq);
102  std::cerr << "\tVariable \"" << key << "\" = \"" << value << "\"" << std::endl;
103  auto vi = _variables.find(key);
104  if (vi != _variables.end())
105  vi->second.push_back(value);
106  else {
107  stringlist list;
108  list.push_back(value);
109  _variables[key] = list;
110  }
111  continue;
112  }
113 
114  // Any other line beginning with '#' is a comment
115  if (line[0] == '#')
116  continue;
117 
118  // Lines starting with '-' are EXIF, IPTC, or XMP tags
119  if (line.substr(0, 1) == "-") {
120  int start = line.find_first_not_of(" \t", 1);
121  int eq = line.find_first_of('=', start);
122  int end = line.find_last_not_of(" \t");
123  std::string value_string = line.substr(eq + 1, end - eq);
124 
125  if (line.substr(start, 3) == "XMP") {
126  std::string key_string = line.substr(start, eq - start);
127 
128  try {
129  Exiv2::XmpKey key = xmp_key_read(key_string);
130  Exiv2::Value::AutoPtr value = Exiv2::Value::create(Exiv2::xmpText);
131  value->read(value_string);
132  std::cerr << "\tXMP \"" << key << "\" = \"" << *value << "\"" << std::endl;
133  _XMPtags.add(key, value.get());
134  } catch (Exiv2::Error& e) {
135  std::cerr << "** XMP key \"" << key_string << "\" not accepted **" << std::endl;
136  }
137  } else if (line.substr(start, 4) == "IPTC") {
138  std::string key_string = line.substr(start, eq - start);
139 
140  try {
141  Exiv2::IptcKey key = iptc_key_read(key_string);
142  Exiv2::Value::AutoPtr value = Exiv2::Value::create(Exiv2::asciiString);
143  value->read(value_string);
144  std::cerr << "\tIPTC \"" << key << "\" = \"" << *value << "\"" << std::endl;
145  _IPTCtags.add(key, value.get());
146  } catch (Exiv2::Error& e) {
147  std::cerr << "** IPTC key \"" << key_string << "\" not accepted **" << std::endl;
148  }
149  } else {
150  std::string key_string = line.substr(start, eq - start);
151 
152  try {
153  Exiv2::ExifKey key = exif_key_read(key_string);
154  Exiv2::Value::AutoPtr value = exif_value_read(key, value_string);
155 
156  std::cerr << "\tEXIF \"" << key << "\" (" << key.defaultTypeId() << ") = \"" << *value << "\"" << std::endl;
157  _EXIFtags.add(key, value.get());
158  } catch (Exiv2::Error& e) {
159  std::cerr << "** EXIF key \"" << key_string << "\" not accepted **" << std::endl;
160  }
161  }
162  continue;
163  }
164 
165  std::cerr << "** Unrecognised line \"" << line << "\" in file \"" << filepath.native() << "\" **" << std::endl;
166  }
167  fin.close();
168  std::cerr << "Finished loading \"" << filepath.native() << "\"" << std::endl;
169  }

182  void Tags::make_thumbnail(Image::ptr img, const D_thumbnail& dt) {
183 #ifdef HAZ_JPEG
184  double width, height;
185 
186  if (dt.maxwidth() * img->height() < dt.maxheight() * img->width()) {
187  width = dt.maxwidth();
188  height = dt.maxwidth() * img->height() / img->width();
189  } else {
190  width = dt.maxheight() * img->width() / img->height();
191  height = dt.maxheight();
192  }
193 
194  std::cerr << "Making EXIF thumbnail..." << std::endl;
195 
196  Frame frame(width, height, 0, 0, img->width(), img->height());
197  auto thumbimage = frame.crop_resize(img, D_resize::lanczos(3.0));
198 
199  auto dest = std::make_shared<Destination>();
200  dest->set_jpeg(D_JPEG(50, 1, 1, false));
201  dest->set_depth(8);
202 
203  JPEGwriter thumbfile("");
204 
205  CMS::Format dest_format = thumbfile.preferred_format(dest->modify_format(img->format()));
206  CMS::Profile::ptr dest_profile = dest->get_profile(CMS::ColourModel::RGB, "JPEG thumbnail");
207  thumbimage->transform_colour_inplace(dest_profile, dest_format);
208 
209  std::ostringstream oss;
210  thumbfile.write(oss, thumbimage, dest);
211 
212  Exiv2::ExifThumb EXIFthumb(_EXIFtags);
213  EXIFthumb.setJpegThumbnail((unsigned char*)oss.str().data(), oss.str().length());
214 
215  std::cerr << "Done." << std::endl;
216 #endif
217  }

220  Exiv2::URationalValue v;
221  if (img->xres().defined()) {
222  v = closest_Rational<unsigned int, Exiv2::URational>(img->xres());

228  }
