27  void PNGreader_cb::info(png_structp png, png_infop info) {
28  png_set_gamma(png, 1.0, 1.0);
29 #ifdef PNG_READ_ALPHA_MODE_SUPPORTED
30  png_set_alpha_mode(png, PNG_ALPHA_PNG, 1.0);
31 #endif
32  png_set_packing(png);
33  png_set_swap(png);
34  png_read_update_info(png, info);
35 
36  png_uint_32 width, height;
37  int bit_depth, colour_type;
38  png_get_IHDR(png, info, &width, &height, &bit_depth, &colour_type, NULL, NULL, NULL);
39  std::cerr << "\t" << width << "×" << height << ", " << bit_depth << " bpp, type " << colour_type << "." << std::endl;
40 
41  _destination->set_depth(bit_depth);
42 
43  CMS::Format format;
44  switch (bit_depth >> 3) {
45  case 1: format.set_8bit();
46  break;
47 
48  case 2: format.set_16bit();
49  break;
50 
51  case 4: format.set_32bit();
52  break;
53 
54  default:
55  std::cerr << "** Unknown depth " << (bit_depth >> 3) << " **" << std::endl;
56  throw LibraryError("libTIFF", "depth");
57  }
58 
59  switch (colour_type) {
60  case PNG_COLOR_TYPE_GRAY_ALPHA:
61  format.set_extra_channels(1);
62  format.set_premult_alpha();
63  case PNG_COLOR_TYPE_GRAY:
64  format.set_colour_model(CMS::ColourModel::Greyscale);
65  break;
66 
67  case PNG_COLOR_TYPE_RGB_ALPHA:
68  format.set_extra_channels(1);
69  format.set_premult_alpha();
70  case PNG_COLOR_TYPE_RGB:
71  format.set_colour_model(CMS::ColourModel::RGB);
72  break;
73 
74  default:
75  std::cerr << "** unsupported PNG colour type " << colour_type << " **" << std::endl;
76  exit(1);
77  }
78  _image = std::make_shared<Image>(width, height, format);
79 
80  {
81  png_uint_32 xres, yres;
82  int unit_type;
83  if (png_get_pHYs(png, info, &xres, &yres, &unit_type)) {
84  switch (unit_type) {
85  case PNG_RESOLUTION_METER:
86  _image->set_resolution(xres * 0.0254, yres * 0.0254);
87  break;
88 
89  case PNG_RESOLUTION_UNKNOWN:
90  break;
91 
92  default:
93  std::cerr << "** unknown unit type " << unit_type << " **" << std::endl;
94  }
95  if (_image->xres().defined() && _image->yres().defined())
96  std::cerr << "\tImage has resolution of " << _image->xres() << "×" << _image->yres() << " PPI." << std::endl;
97  }
98  }
99 
100  if (png_get_valid(png, info, PNG_INFO_iCCP)) {
101  std::cerr << "\tImage has iCCP chunk." << std::endl;
102  png_charp profile_name;
103  int compression_type;
104  png_bytep profile_data;
105  png_uint_32 profile_len;
106 #if PNG_LIBPNG_VER < 10500
107  if (png_get_iCCP(png, info, &profile_name, &compression_type, (png_charpp)&profile_data, &profile_len) == PNG_INFO_iCCP) {
108 #else
109  if (png_get_iCCP(png, info, &profile_name, &compression_type, &profile_data, &profile_len) == PNG_INFO_iCCP) {
110 #endif
111  std::cerr << "\tLoading ICC profile \"" << profile_name << "\" from file..." << std::endl;
112  CMS::Profile::ptr profile = std::make_shared<CMS::Profile>(profile_data, profile_len);
113  unsigned char *data_copy = new unsigned char[profile_len];
114  memcpy(data_copy, profile_data, profile_len);
115  _destination->set_profile(profile_name, data_copy, profile_len);
116  _image->set_profile(profile);
117  }
118  }
119  }
