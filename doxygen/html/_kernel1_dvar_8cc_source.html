48  void Kernel1Dvar::build(double from_start, double from_size, unsigned int from_max) throw(DestinationError) {
49  _scale = from_size / _to_size;
50  double range, norm_fact;
51  if (_scale < 1.0) {
52  range = this->range();
53  norm_fact = 1.0;
54  } else {
55  range = this->range() * _scale;
56  norm_fact = this->range() / ceil(range);
57  }
58 
59 #pragma omp parallel for schedule(dynamic, 1)
60  for (unsigned int i = 0; i < _to_size_i; i++) {
61  double centre = from_start + (i * _scale);
62  unsigned int left = floor(centre - range);
63  if (range > centre)
64  left = 0;
65  unsigned int right = ceil(centre + range);
66  if (right >= from_max)
67  right = from_max - 1;
68  _size[i] = right + 1 - left;
69  _start[i] = left;
70  _weights[i] = (SAMPLE*)malloc(_size[i] * sizeof(SAMPLE));
71  unsigned int k = 0;
72  for (unsigned int j = left; j <= right; j++, k++)
73  _weights[i][k] = this->eval((centre - j) * norm_fact);
74 
75  // normalize the filter's weight's so the sum equals to 1.0, very important for avoiding box type of artifacts
76  unsigned int max = _size[i];
77  SAMPLE tot = 0.0;
78  for (unsigned int k = 0; k < max; k++)
79  tot += _weights[i][k];
80  if (fabs(tot) > 1e-5) {
81  tot = 1.0 / tot;
82  for (unsigned int k = 0; k < max; k++)
83  _weights[i][k] *= tot;
84  }
85  }

126  void Kernel1Dvar::convolve_h_type_channels(Image::ptr src, Image::ptr dest, bool can_free) {
127 #pragma omp parallel
128  {
129 #pragma omp master
130  {
131  std::cerr << "Convolving image horizontally " << src->width() << " => "
132  << std::setprecision(2) << std::fixed << dest->width()
133  << " using " << omp_get_num_threads() << " threads..." << std::endl;
134  }
135  }
136 
137 #pragma omp parallel for schedule(dynamic, 1)
138  for (unsigned int y = 0; y < src->height(); y++) {
139  dest->check_rowdata_alloc(y);
140  T *out = dest->row<T>(y);
141  SAMPLE temp[channels];
142 
143  for (unsigned int nx = 0; nx < dest->width(); nx++) {
144  for (unsigned char c = 0; c < channels; c++)
145  temp[c] = 0;
146  const SAMPLE *weight = _weights[nx];
147  const T *in = src->at<T>(_start[nx], y);
148  for (unsigned int j = _size[nx]; j; j--, weight++) {
149  for (unsigned char c = 0; c < channels; c++, in++)
150  temp[c] += (*in) * (*weight);
151  }
152  for (unsigned char c = 0; c < channels; c++, out++)
153  *out = limitval<T>(temp[c]);
154  }
155 
156  if (can_free)
157  src->free_row(y);
158  if (omp_get_thread_num() == 0)
159  std::cerr << "\r\tConvolved " << y + 1 << " of " << src->height() << " rows";
160  }
161  std::cerr << "\r\tConvolved " << src->height() << " of " << src->height() << " rows." << std::endl;
162  }

272 #pragma omp parallel
273  {
274 #pragma omp master
275  {
276  std::cerr << "Convolving image vertically " << src->height() << " => "

279  }
280  }
281 
282  int *row_needs;
283  if (can_free) {
284  row_needs = (int*)malloc(src->height() * sizeof(int));
285  int num_threads = omp_get_num_threads();
286  for (unsigned int y = 0; y < src->height(); y++)
287  row_needs[y] = num_threads;
288  }
289  unsigned int next_freed = 0;
290  omp_lock_t freed_lock;
291  omp_init_lock(&freed_lock);
292 #pragma omp parallel for schedule(dynamic, 1)
293  for (unsigned int ny = 0; ny < dest->height(); ny++) {
294  unsigned int max = _size[ny];
295  unsigned int ystart = _start[ny];
296 
297  dest->check_rowdata_alloc(ny);
298  T *out = dest->row<T>(ny);
299  SAMPLE temp[channels];
300  for (unsigned int x = 0; x < src->width(); x++) {
301  for (unsigned char c = 0; c < channels; c++)
302  temp[c] = 0;
303 
304  const SAMPLE *weight = _weights[ny];
305  for (unsigned int j = 0; j < max; j++, weight++) {
306  T *in = src->at<T>(x, ystart + j);
307  for (unsigned char c = 0; c < channels; c++, in++)
308  temp[c] += (*in) * (*weight);
309  }
310 
311  for (unsigned char c = 0; c < channels; c++, out++)
312  *out = limitval<T>(temp[c]);
313  }
314 
315  if (can_free && (((ny < _to_size_i - 1) && (_start[ny + 1] > ystart)) || (ny == _to_size_i - 1))) {
316  omp_set_lock(&freed_lock);
317  row_needs[ystart]--;
318  for (;row_needs[next_freed] == 0; next_freed++)
319  src->free_row(next_freed);
320  omp_unset_lock(&freed_lock);
321  }
322  if (omp_get_thread_num() == 0)
323  std::cerr << "\r\tConvolved " << ny + 1 << " of " << _to_size_i << " rows";
324  }
325  std::cerr << "\r\tConvolved " << _to_size_i << " of " << _to_size_i << " rows." << std::endl;
326  if (can_free) {
327  free(row_needs);
328  for (; next_freed < src->height(); next_freed++)
329  src->free_row(next_freed);
330  }
331  omp_destroy_lock(&freed_lock);
332  }

457  SAMPLE Lanczos::eval(double x) const throw(Uninitialised) {
458  if (!_radius.defined())
459  throw Uninitialised("Lanczos", "resize.radius");
460 
461  if (fabs(x) < 1e-6)
462  return 1.0;
463  double pix = M_PI * x;
464  return (_radius * sin(pix) * sin(pix * _r_radius)) / (sqr(M_PI) * sqr(x));
