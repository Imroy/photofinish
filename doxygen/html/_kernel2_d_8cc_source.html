68  void Kernel2D::do_convolve(Image::ptr src, Image::ptr dest, bool can_free) {
69  cmsUInt32Number type = src->type();
70  unsigned int pixel_size = T_CHANNELS(type) + T_EXTRA(type);
71 
72  int *row_needs;
73  if (can_free) {
74  int num_threads = omp_get_num_threads();
75  row_needs = (int*)malloc(src->height() * sizeof(int));
76  for (unsigned int y = 0; y < src->height(); y++)
77  row_needs[y] = num_threads;
78  }
79  unsigned int next_freed = 0;
80  omp_lock_t freed_lock;
81  omp_init_lock(&freed_lock);
82 #pragma omp parallel for schedule(dynamic, 1)
83  for (unsigned int y = 0; y < src->height(); y++) {
84  T *outp = (T*)dest->row(y);
85  short unsigned int ky_start = y < _centrey ? _centrey - y : 0;
86  short unsigned int ky_end = y > src->height() - _height + _centrey ? src->height() + _centrey - y : _height;
87 
88  for (unsigned int x = 0; x < src->width(); x++, outp += pixel_size) {
89  short unsigned int kx_start = x < _centrex ? _centrex - x : 0;
90  short unsigned int kx_end = x > src->width() - _width + _centrex ? src->width() + _centrex - x : _width;
91 
92  double weight = 0;
93  for (unsigned char c = 0; c < T_CHANNELS(type); c++)
94  outp[c] = 0;
95 
96  for (short unsigned int ky = ky_start; ky < ky_end; ky++) {
97  const SAMPLE *kp = &at(kx_start, ky);
98  T *inp = (T*)src->at(x + kx_start - _centrex, y + ky - _centrey);
99  for (short unsigned int kx = kx_start; kx < kx_end; kx++, kp++, inp += T_EXTRA(type)) {
100  weight += *kp;
101  for (unsigned char c = 0; c < T_CHANNELS(type); c++, inp++)
102  outp[c] += (*inp) * (*kp);
103  }
104  }
105  if (fabs(weight) > 1e-5) {
106  weight = 1.0 / weight;
107  for (unsigned char c = 0; c < T_CHANNELS(type); c++)
108  outp[c] *= weight;
109  }
110  }
111  if (can_free && (y > _centrey)) {
112  omp_set_lock(&freed_lock);
113  row_needs[y + ky_start - _centrey]--;
114  for (;row_needs[next_freed] == 0; next_freed++)
115  src->free_row(next_freed);
116  omp_unset_lock(&freed_lock);
117  }
118 
119  if (omp_get_thread_num() == 0)
120  std::cerr << "\r\tConvolved " << y + 1 << " of " << src->height() << " rows";
121  }
122  std::cerr << "\r\tConvolved " << src->height() << " of " << src->height() << " rows." << std::endl;
123  if (can_free) {
124  free(row_needs);
125  for (; next_freed < src->height(); next_freed++)
126  src->free_row(next_freed);
127  }
128  }

130  Image::ptr Kernel2D::convolve(Image::ptr img, bool can_free) {
131 #pragma omp parallel
132  {
133 #pragma omp master
134  {
135  std::cerr << "Convolving " << img->width() << "×" << img->height()
136  << " image with " << _width << "×" << _height
137  << " kernel using " << omp_get_num_threads() << " threads..." << std::endl;
138  }
139  }
140  auto out = std::make_shared<Image>(img->width(), img->height(), img->type());
141 
142  if (img->xres().defined())
143  out->set_xres(img->xres());
144  if (img->yres().defined())
145  out->set_yres(img->yres());
146 
147  switch (T_BYTES_REAL(img->type())) {
148  case 1:
149  do_convolve<unsigned char>(img, out, can_free);
150  break;
151 
152  case 2:
153  do_convolve<short unsigned int>(img, out, can_free);
154  break;
155 
156  case 4:
157  if (T_FLOAT(img->type()))
158  do_convolve<float>(img, out, can_free);
159  else
160  do_convolve<unsigned int>(img, out, can_free);
161  break;
162 
163  case 8:
164  if (T_FLOAT(img->type()))
165  do_convolve<double>(img, out, can_free);
166  else
167  do_convolve<unsigned long>(img, out, can_free);
168  break;
169 
170  }
171 
172  return out;
173  }

180  GaussianSharpen::GaussianSharpen(const D_sharpen& ds) :
181  Kernel2D(1 + (2 * ceil(ds.radius())), ceil(ds.radius())),

187  throw Uninitialised("GaussianSharpen", "sharpen.radius");
188

190  throw Uninitialised("GaussianSharpen", "sharpen.sigma");
191 
192  SAMPLE total = 0;
193 #pragma omp parallel for schedule(dynamic, 1) shared(total)
194  for (short unsigned int y = 0; y < _height; y++)
195  for (short unsigned int x = 0; x < _width; x++)
196  total += at(x, y) = -exp((sqr((int)x - _centrex) + sqr((int)y - _centrey)) / (-2.0 * _safe_sigma_sqr));
