266  template <typename Num_type, typename R_type>
267  Exiv2::Value::AutoPtr parse_Rational(std::string s) {
268  size_t slash = s.find_first_of('/');
269  Exiv2::ValueType<R_type> *rv;
270  if (slash == std::string::npos)
271  rv = &closest_Rational<Num_type, R_type>(boost::lexical_cast<double>(s));
272  else
273  rv = new Exiv2::ValueType<R_type>(R_type(boost::lexical_cast<Num_type>(s.substr(0, slash)),
274  boost::lexical_cast<Num_type>(s.substr(slash + 1, s.length() - slash - 1))));
275 
276  return Exiv2::Value::AutoPtr(rv);
277  }

280  Exiv2::Value::AutoPtr exif_value_read(Exiv2::ExifKey key, std::string value_string) {
281  Exiv2::TypeId type = key.defaultTypeId();
282  if (type == Exiv2::unsignedRational)
283  return parse_Rational<unsigned int, Exiv2::URational>(value_string);
284  if (type == Exiv2::signedRational)
285  return parse_Rational<signed int, Exiv2::Rational>(value_string);

294  for (auto i : hi->second)

299  }
300 
301  Exiv2::Value::AutoPtr value = Exiv2::Value::create(type);
302  value->read(value_string);

304  return value;
