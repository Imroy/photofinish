78  void jpeg_write_profile(jpeg_compress_struct* cinfo, unsigned char *data, unsigned int size) {
79  unsigned char num_markers = ceil(size / 65519.0); // max 65533 bytes in a marker, minus 14 bytes for the ICC overhead

82  unsigned int data_left = size;
