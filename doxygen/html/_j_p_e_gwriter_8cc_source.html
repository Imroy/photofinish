63  void JPEGwriter::write(std::ostream& os, Image::ptr img, Destination::ptr dest, bool can_free) {
64  jpeg_compress_struct *cinfo = new jpeg_compress_struct;
65  jpeg_create_compress(cinfo);
66  jpeg_error_mgr jerr;
67  cinfo->err = jpeg_std_error(&jerr);
68  jerr.error_exit = jpeg_error_exit;
69 
70  jpeg_ostream_dest(cinfo, &os);
71 
72  cinfo->image_width = img->width();
73  cinfo->image_height = img->height();
74 
75  CMS::Format format = img->format();
76  if (format.bytes_per_channel() != 1)
77  throw cmsTypeError("Not 8-bit", format);
78 
79  switch (format.colour_model()) {
80  case CMS::ColourModel::Greyscale:
81  cinfo->in_color_space = JCS_GRAYSCALE;
82  break;
83 
84  case CMS::ColourModel::RGB:
85  cinfo->in_color_space = JCS_RGB;
86  break;
87 
88  case CMS::ColourModel::YCbCr:
89  cinfo->in_color_space = JCS_YCbCr;
90  break;
91 
92  case CMS::ColourModel::CMYK:
93  cinfo->in_color_space = JCS_CMYK;
94  break;
95 
96  default:
97  break;
98  }
99  cinfo->input_components = format.channels();
100 
101  jpeg_set_defaults(cinfo);
102  {
103  // Default values
104  int quality = 95;
105  int sample_h = 2, sample_v = 2;
106  bool progressive = true;
107 
108  if (dest->jpeg().defined()) {
109  const D_JPEG jpeg = dest->jpeg();
110  if (jpeg.quality().defined())
111  quality = jpeg.quality();
112  if (jpeg.progressive().defined())
113  progressive = jpeg.progressive();
114  if (jpeg.sample().defined() && (format.colour_model() == CMS::ColourModel::RGB)) {
115  sample_h = jpeg.sample()->first;
116  sample_v = jpeg.sample()->second;
117  }
118  }
119 
120  std::cerr << "\tJPEG quality of " << quality << "." << std::endl;
121  jpeg_set_quality(cinfo, quality, TRUE);
122 
123  if (progressive) {
124  std::cerr << "\tProgressive JPEG." << std::endl;
125  switch (format.colour_model()) {
126  case CMS::ColourModel::Greyscale:
127  jpegfile_scan_greyscale(cinfo);
128  break;
129 
130  case CMS::ColourModel::RGB:
131  jpegfile_scan_RGB(cinfo);
132  break;
133 
134  default:
135  break;
136  }
137  }
138 
139  if (format.colour_model() != CMS::ColourModel::Greyscale) {
140  std::cerr << "\tJPEG chroma sub-sampling of " << sample_h << "×" << sample_v << "." << std::endl;
141  cinfo->comp_info[0].h_samp_factor = sample_h;
142  cinfo->comp_info[0].v_samp_factor = sample_v;
143  }
144  }
145 
146  cinfo->density_unit = 1; // PPI
147  cinfo->X_density = round(img->xres());
148  cinfo->Y_density = round(img->yres());
149 
150  cinfo->dct_method = JDCT_FLOAT;
151 
152  jpeg_start_compress(cinfo, TRUE);
153 
154  if (img->has_profile()) {
155  unsigned char *profile_data;
156  unsigned int profile_len;
157  img->profile()->save_to_mem(profile_data, profile_len);
158 
159  if (profile_len > 0) {
160  if (profile_len > 255 * 65519)
161  std::cerr << "** Profile is too big to fit in APP2 markers! **" << std::endl;
162  else {
163  jpeg_write_profile(cinfo, (unsigned char*)profile_data, profile_len);
164  delete [] profile_data;
165  }
166  }
167  }
168 
169  JSAMPROW jpeg_row[1];
170  std::cerr << "\tWriting " << img->width() << "×" << img->height() << " JPEG image..." << std::endl;
171  while (cinfo->next_scanline < cinfo->image_height) {
172  unsigned int y = cinfo->next_scanline;
173  jpeg_row[0] = img->row<unsigned char>(y);
174  jpeg_write_scanlines(cinfo, jpeg_row, 1);
175 
176  if (can_free)
177  img->free_row(y);
178 
179  std::cerr << "\r\tWritten " << y + 1 << " of " << img->height() << " rows";
180  }
181  std::cerr << "\r\tWritten " << img->height() << " of " << img->height() << " rows." << std::endl;
182 
183  jpeg_finish_compress(cinfo);
184  jpeg_ostream_dest_free(cinfo);
185  jpeg_destroy_compress(cinfo);
186  delete cinfo;
187  }
