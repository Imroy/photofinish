36  Image::ptr TIFFreader::read(Destination::ptr dest) {
37  if (_is_open)
38  throw FileOpenError("already open");
39  _is_open = true;
40 
41  std::cerr << "Opening file " << _filepath << "..." << std::endl;
42  fs::ifstream fb(_filepath, std::ios_base::in);
43  if (fb.fail())
44  throw FileOpenError(_filepath.native());
45 
46  TIFF *tiff = TIFFStreamOpen("", &fb);
47  if (tiff == NULL)
48  throw FileOpenError(_filepath.native());
49 
50  int rc;
51  uint32 width, height;
52  TIFFcheck(GetField(tiff, TIFFTAG_IMAGEWIDTH, &width));
53  TIFFcheck(GetField(tiff, TIFFTAG_IMAGELENGTH, &height));
54  std::cerr << "\tImage is " << width << "×" << height << std::endl;
55 
56  uint16 bit_depth, channels, photometric;
57  TIFFcheck(GetField(tiff, TIFFTAG_BITSPERSAMPLE, &bit_depth));
58  dest->set_depth(bit_depth);
59  std::cerr << "\tImage has a depth of " << bit_depth << std::endl;
60 
61  TIFFcheck(GetField(tiff, TIFFTAG_SAMPLESPERPIXEL, &channels));
62 
63  CMS::Format format;
64  switch (bit_depth >> 3) {
65  case 1: format.set_8bit();
66  break;
67 
68  case 2: format.set_16bit();
69  break;
70 
71  case 4: format.set_32bit();
72  break;
73 
74  default:
75  std::cerr << "** Unknown depth " << (bit_depth >> 3) << " **" << std::endl;
76  throw LibraryError("libTIFF", "depth");
77  }
78 
79  {
80  uint16 extra_count, *extra_types;
81  if (TIFFGetField(tiff, TIFFTAG_EXTRASAMPLES, &extra_count, &extra_types) == 1) {
82  format.set_extra_channels(extra_count & 0x07);
83  for (int i = 0; i < extra_count; i++) {
84  std::cerr << "\tImage has an ";
85  switch (extra_types[i]) {
86  case EXTRASAMPLE_UNSPECIFIED: std::cerr << "unspecified ";
87  break;
88  case EXTRASAMPLE_ASSOCALPHA: std::cerr << "associated alpha ";
89  format.set_premult_alpha();
90  break;
91  case EXTRASAMPLE_UNASSALPHA: std::cerr << "unassociated alpha ";
92  format.unset_premult_alpha();
93  break;
94  default: std::cerr << "unknown ";
95  }
96  std::cerr << "extra channel." << std::endl;
97  }
98  channels -= extra_count;
99  }
100  }
101  std::cerr << "\tImage has " << channels << " channels." << std::endl;
102 
103  TIFFcheck(GetField(tiff, TIFFTAG_PHOTOMETRIC, &photometric));
104  switch (photometric) {
105  case PHOTOMETRIC_MINISWHITE:
106  format.set_vanilla();
107  case PHOTOMETRIC_MINISBLACK:
108  format.set_colour_model(CMS::ColourModel::Greyscale, channels);
109  break;
110 
111  case PHOTOMETRIC_RGB:
112  format.set_colour_model(CMS::ColourModel::RGB, channels);
113  break;
114 
115  case PHOTOMETRIC_SEPARATED:
116  format.set_colour_model(CMS::ColourModel::CMYK, channels);
117  break;
118 
119  default:
120  std::cerr << "** unsupported TIFF photometric interpretation " << photometric << " **" << std::endl;
121  exit(1);
122  }
123  auto img = std::make_shared<Image>(width, height, format);
124 
125  {
126  float xres, yres;
127  short unsigned int resunit;
128  if ((TIFFGetField(tiff, TIFFTAG_XRESOLUTION, &xres) == 1)
129  && (TIFFGetField(tiff, TIFFTAG_YRESOLUTION, &yres) == 1)
130  && (TIFFGetField(tiff, TIFFTAG_RESOLUTIONUNIT, &resunit) == 1)) {
131  switch (resunit) {
132  case RESUNIT_INCH:
133  img->set_resolution(xres, yres);
134  break;
135  case RESUNIT_CENTIMETER:
136  img->set_resolution((double)xres * 2.54, (double)yres * 2.54);
137  break;
138  default:
139  std::cerr << "** unknown resolution unit " << resunit << " **" << std::endl;
140  }
141  if (img->xres().defined() && img->yres().defined())
142  std::cerr << "\tImage has resolution of " << img->xres() << "×" << img->yres() << " PPI." << std::endl;
143  }
144  }
145 
146  {
147  uint32 profile_len;
148  unsigned char *profile_data;
149  if (TIFFGetField(tiff, TIFFTAG_ICCPROFILE, &profile_len, &profile_data) == 1) {
150  CMS::Profile::ptr profile = std::make_shared<CMS::Profile>(profile_data, profile_len);
151  unsigned char *data_copy = new unsigned char[profile_len];
152  memcpy(data_copy, profile_data, profile_len);
153 
154  std::string profile_name = profile->description("en", "");
155  if (profile_name.length() > 0)
156  dest->set_profile(profile_name, data_copy, profile_len);
157  else
158  dest->set_profile("TIFFTAG_ICCPROFILE", data_copy, profile_len);
159  img->set_profile(profile);
160  std::cerr << "\tRead embedded profile \"" << dest->profile()->name().get() << "\" (" << profile_len << " bytes)" << std::endl;
161  }
162  }
163 
164  std::cerr << "\tReading TIFF image..." << std::endl;
165  for (unsigned int y = 0; y < height; y++) {
166  img->check_rowdata_alloc(y);
167  TIFFcheck(ReadScanline(tiff, img->row(y), y));
168  std::cerr << "\r\tRead " << (y + 1) << " of " << height << " rows";
169  }
170  std::cerr << "\r\tRead " << height << " of " << height << " rows." << std::endl;
171 
172  TIFFClose(tiff);
173  fb.close();
174  _is_open = false;
175 
176  std::cerr << "\tExtracting tags..." << std::endl;
177  extract_tags(img);
178 
179  std::cerr << "Done." << std::endl;
180  return img;
181  }
