100  Frame::ptr Destination::best_frame(Image::ptr img) {
101  if (_targets.size() == 0)
102  throw NoTargets(_name);
103 
104  std::cerr << "Finding best target from \"" << _name << "\" to fit image " << img->width() << "x" << img->height() << "..." << std::endl;
105  CropSolver solver(_variables);
106  Frame::ptr best_frame;
107  double best_waste = 0;
108  for (auto ti : _targets) {
109  auto target = ti.second;
110  std::cerr << "\tTarget \"" << target->name() << "\" (" << target->width() << "×" << target->height() << "):" << std::endl;
111 
112  if ((target->width() > img->width()) && (target->height() > img->height())) {
113  std::cerr << "\tSkipping because the target is larger than the original image in both dimensions." << std::endl;
114  continue;
115  }
116 
117  if (target->width() * target->height() > img->width() * img->height()) {
118  std::cerr << "\tSkipping because the target has more pixels than the original image." << std::endl;
119  continue;
120  }
121 
122  std::cerr << std::setprecision(2) << std::fixed;
123  auto frame = solver.solve(img, target);
124 
125  if ((target->width() > frame->crop_w()) && (target->height() > frame->crop_h())) {
126  std::cerr << "\tSkipping because the target is larger than the cropped image in both dimensions." << std::endl;
127  continue;
128  }
129 
130  if (target->width() * target->height() > frame->crop_w() * frame->crop_h()) {
131  std::cerr << "\tSkipping because the target has more pixels than the cropped image." << std::endl;
132  continue;
133  }
134 
135  double waste = frame->waste(img);
136 
137  std::cerr << "\tWaste from ("
138  << frame->crop_x() << ", " << frame->crop_y() << ") + ("
139  << frame->crop_w() << "×" << frame->crop_h() << ") = "
140  << waste << "." << std::endl;
141  if ((!best_frame) || (waste < best_waste)) {
142  best_frame.swap(frame);
143  best_waste = waste;
144  }
145  }
146 
147  if (!best_frame)
148  throw NoResults("Destination", "best_frame");
149 
150  std::cerr << "Least waste was from frame \"" << best_frame->name() << "\" = " << best_waste << "." << std::endl;
151  return best_frame;
152  }

154  CMS::Format Destination::modify_format(CMS::Format format) {
155  if (this->forcergb().defined() && this->forcergb()) {
156  std::cerr << "Forcing RGB..." << std::endl;
157  format.set_colour_model(CMS::ColourModel::RGB);
158  }
159 
160  if (this->forcegrey().defined() && this->forcegrey()) {
161  std::cerr << "Forcing greyscale..." << std::endl;
162  format.set_colour_model(CMS::ColourModel::Greyscale);
163  }
164 
165  if (this->depth().defined()) {
166  std::cerr << "Changing depth to " << (this->depth() >> 3) << " bytes..." << std::endl;
167  switch (this->depth() >> 3) {
168  case 1: format.set_8bit();
169  break;
170 
171  case 2: format.set_16bit();
172  break;
173 
174  case 4: format.set_32bit();
175  break;
176 
177  default:
178  std::cerr << "Invalid depth." << std::endl;
179  format.set_8bit();
180  break;
181  }
182  } else {
183  std::cerr << "Changing depth to 1 byte..." << std::endl;
184  format.set_8bit();
185  }
186 
187  return format;
188  }

207  void Destination::read_config(const YAML::Node& node) {
208  if (node["name"])
209  _name = node["name"].as<std::string>();
210 
211  if (node["dir"])
212  _dir = node["dir"].as<std::string>();
213 
214  if (node["size"])
215  _size = node["size"].as<double>();
216 
217  if (node["format"])
218  _format = node["format"].as<std::string>();
219 
220  if (node["depth"])
221  _depth = node["depth"].as<int>();
222 
223  if (node["noresize"])
224  _noresize = node["noresize"].as<bool>();
225 
226  if (node["forcergb"])
227  _forcergb = node["forcergb"].as<bool>();
228 
229  if (node["forcegrey"])
230  _forcegrey = node["forcegrey"].as<bool>();
231 
232  if (node["intent"]) {
233  std::string intent;
234  intent = node["intent"].as<std::string>();
235  if (boost::iequals(intent.substr(0, 10), "perceptual"))
236  _intent = CMS::Intent::Perceptual;
237  else if (boost::iequals(intent.substr(0, 8), "relative"))
238  _intent = CMS::Intent::Relative_colormetric;
239  else if (boost::iequals(intent.substr(0, 10), "saturation"))
240  _intent = CMS::Intent::Saturation;
241  else if (boost::iequals(intent.substr(0, 8), "absolute"))
242  _intent = CMS::Intent::Absolute_colormetric;
243  else
244  _intent = CMS::Intent::Perceptual;
245  }
246 
247  if (node["sharpen"])
248  _sharpen.read_config(node["sharpen"]);
249 
250  if (node["resize"])
251  _resize.read_config(node["resize"]);
252 
253  if (node["jpeg"])
254  _jpeg.read_config(node["jpeg"]);
255 
256  if (node["png"])
257  _png.read_config(node["png"]);
258 
259  if (node["tiff"])
260  _tiff.read_config(node["tiff"]);
261 
262  if (node["jp2"])
263  _jp2.read_config(node["jp2"]);
264 
265  if (node["webp"])
266  _webp.read_config(node["webp"]);
267 
268  if (node["jxr"])
269  _jxr.read_config(node["jxr"]);
270 
271  if (node["profile"]) {
272  _profile = std::make_shared<D_profile>();
273  _profile->read_config(node["profile"]);
274  }
275 
276  if (node["thumbnail"])
277  _thumbnail.read_config(node["thumbnail"]);
278 
279  if (node["targets"]) {
280  YAML::Node targets = node["targets"];
281  for(auto ti = targets.begin(); ti != targets.end(); ti++) {
282  std::string tname = ti->first.as<std::string>();
283  auto target = std::make_shared<D_target>(tname);
284  target->read_config(ti->second);
285  _targets[tname] = target;
286  }
287  }
