96  Frame::ptr Destination::best_frame(Image::ptr img) {
97  if (_targets.size() == 0)
98  throw NoTargets(_name);
99 
100  std::cerr << "Finding best target from \"" << _name << "\" to fit image " << img->width() << "x" << img->height() << "..." << std::endl;
101  CropSolver solver(_variables);
102  Frame::ptr best_frame;
103  double best_waste = 0;
104  for (std::map<std::string, D_target::ptr>::const_iterator ti = _targets.begin(); ti != _targets.end(); ti++) {
105  D_target::ptr target = ti->second;
106  std::cerr << "\tTarget \"" << target->name() << "\" (" << target->width() << "×" << target->height() << "):" << std::endl;
107 
108  if ((target->width() > img->width()) && (target->height() > img->height())) {
109  std::cerr << "\tSkipping because the target is larger than the original image in both dimensions." << std::endl;
110  continue;
111  }
112 
113  if (target->width() * target->height() > img->width() * img->height()) {
114  std::cerr << "\tSkipping because the target has more pixels than the original image." << std::endl;
115  continue;
116  }
117 
118  Frame::ptr frame = solver.solve(img, target);
119 
120  if ((target->width() > frame->crop_w()) && (target->height() > frame->crop_h())) {
121  std::cerr << "\tSkipping because the target is larger than the cropped image in both dimensions." << std::endl;
122  continue;
123  }
124 
125  if (target->width() * target->height() > frame->crop_w() * frame->crop_h()) {
126  std::cerr << "\tSkipping because the target has more pixels than the cropped image." << std::endl;
127  continue;
128  }
129 
130  double waste = frame->waste(img);
131 
132  std::cerr << "\tWaste from ("
133  << std::setprecision(2) << std::fixed << frame->crop_x() << ", " << frame->crop_y() << ") + ("
134  << frame->crop_w() << "×" << frame->crop_h() << ") = "
135  << waste << "." << std::endl;
136  if ((!best_frame) || (waste < best_waste)) {
137  best_frame.swap(frame);
138  best_waste = waste;
139  }
140  }
141 
142  if (!best_frame)
143  throw NoResults("Destination", "best_frame");
144 
145  std::cerr << "Least waste was from frame \"" << best_frame->name() << "\" = " << best_waste << "." << std::endl;
146  return best_frame;
147  }

172  if (const YAML::Node *n = node.FindValue("forcegrey"))
173  *n >> d._forcegrey;

175  if (const YAML::Node *n = node.FindValue("intent")) {

183  d._intent = INTENT_SATURATION;

185  d._intent = INTENT_ABSOLUTE_COLORIMETRIC;
186  else
187  d._intent = INTENT_PERCEPTUAL;
188  }
189 
190  if (const YAML::Node *n = node.FindValue("sharpen"))
191  *n >> d._sharpen;
192 
193  if (const YAML::Node *n = node.FindValue("resize"))
194  *n >> d._resize;
195 
196  if (const YAML::Node *n = node.FindValue("jpeg"))
197  *n >> d._jpeg;

201

204

206  *n >> d._jp2;
