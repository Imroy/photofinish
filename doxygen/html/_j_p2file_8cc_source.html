51  template <typename T>
52  inline void* planar_to_packed(unsigned int width, unsigned char channels, opj_image_t* image, unsigned int& index) {
53  T *row = (T*)malloc(width * channels * sizeof(T));
54  T *out = row;
55  for (unsigned int x = 0; x < width; x++, index++)
56  for (unsigned char c = 0; c < channels; c++)
57  *out++ = image->comps[c].data[index];
58  return (void*)row;
59  }

65  for (unsigned int x = 0; x < width; x++, index++)
66  for (unsigned char c = 0; c < channels; c++)

70  Image::ptr JP2file::read(Destination::ptr dest) const {
71  opj_event_mgr_t event_mgr;
72  memset(&event_mgr, 0, sizeof(opj_event_mgr_t));
73  event_mgr.error_handler = error_callback;
74  event_mgr.warning_handler = warning_callback;
75  event_mgr.info_handler = info_callback;
76 
77  opj_dparameters_t parameters;
78  opj_set_default_decoder_parameters(&parameters);
79 
80  std::cerr << "Opening file " << _filepath << "..." << std::endl;
81  fs::ifstream ifs(_filepath, std::ios_base::in);
82  if (ifs.fail())
83  throw FileOpenError(_filepath.native());
84 
85  unsigned int file_size = fs::file_size(_filepath);
86  unsigned char *src = (unsigned char *)malloc(file_size);
87  ifs.read((char*)src, file_size);
88 
89  opj_dinfo_t *dinfo = opj_create_decompress(CODEC_JP2);
90  opj_set_event_mgr((opj_common_ptr)dinfo, &event_mgr, (void*)this);
91  opj_setup_decoder(dinfo, &parameters);
92  opj_cio_t *cio = opj_cio_open((opj_common_ptr)dinfo, src, file_size);
93  opj_image_t *jp2_image = opj_decode(dinfo, cio);
94  if (jp2_image == NULL)
95  throw LibraryError("OpenJPEG", "Could not decode file");
96 
97  opj_cio_close(cio);
98  free(src);
99 
100  // Is this necessary?
101  if (jp2_image->numcomps > 1)
102  for (unsigned char c = 1; c < jp2_image->numcomps; c++) {
103  if (jp2_image->comps[c].dx != jp2_image->comps[0].dx)
104  std::cerr << "** Component " << (int)c << " has a different dx to the first! **" << std::endl;
105  if (jp2_image->comps[c].dy != jp2_image->comps[0].dy)
106  std::cerr << "** Component " << (int)c << " has a different dy to the first! **" << std::endl;
107  if (jp2_image->comps[c].w != jp2_image->comps[0].w)
108  std::cerr << "** Component " << (int)c << " has a different w to the first! **" << std::endl;
109  if (jp2_image->comps[c].h != jp2_image->comps[0].h)
110  std::cerr << "** Component " << (int)c << " has a different h to the first! **" << std::endl;
111  if (jp2_image->comps[c].x0 != jp2_image->comps[0].x0)
112  std::cerr << "** Component " << (int)c << " has a different x0 to the first! **" << std::endl;
113  if (jp2_image->comps[c].y0 != jp2_image->comps[0].y0)
114  std::cerr << "** Component " << (int)c << " has a different y0 to the first! **" << std::endl;
115  if (jp2_image->comps[c].prec != jp2_image->comps[0].prec)
116  std::cerr << "** Component " << (int)c << " has a different prec to the first! **" << std::endl;
117  if (jp2_image->comps[c].bpp != jp2_image->comps[0].bpp)
118  std::cerr << "** Component " << (int)c << " has a different bpp to the first! **" << std::endl;
119  if (jp2_image->comps[c].sgnd != jp2_image->comps[0].sgnd)
120  std::cerr << "** Component " << (int)c << " has a different sgnd to the first! **" << std::endl;
121  }
122 
123  Image::ptr img(new Image(jp2_image->x1 - jp2_image->x0, jp2_image->y1 - jp2_image->y0));
124 
125  int depth = jp2_image->comps[0].prec;
126  cmsUInt32Number cmsType = CHANNELS_SH(jp2_image->numcomps) | BYTES_SH(depth >> 3);
127  switch (jp2_image->color_space) {
128  case CLRSPC_SRGB:
129  cmsType |= COLORSPACE_SH(PT_RGB);
130  break;
131 
132  case CLRSPC_GRAY:
133  img->set_greyscale();
134  cmsType |= COLORSPACE_SH(PT_GRAY);
135  break;
136 
137  case CLRSPC_SYCC:
138  cmsType |= COLORSPACE_SH(PT_YUV);
139  break;
140 
141  default:
142  std::cerr << "** Unknown colour space " << jp2_image->color_space << " **" << std::endl;
143  throw LibraryError("OpenJPEG", "color_space");
144  }
145  dest->set_depth(depth);
146 
147  cmsHPROFILE profile;
148  if (jp2_image->icc_profile_buf != NULL) {
149  profile = cmsOpenProfileFromMem(jp2_image->icc_profile_buf, jp2_image->icc_profile_len);
150  void *data_copy = malloc(jp2_image->icc_profile_len);
151  memcpy(data_copy, jp2_image->icc_profile_buf, jp2_image->icc_profile_len);
152  char *profile_name = NULL;
153  unsigned int profile_name_len;
154  if ((profile_name_len = cmsGetProfileInfoASCII(profile, cmsInfoDescription, "en", cmsNoCountry, NULL, 0)) > 0) {
155  profile_name = (char*)malloc(profile_name_len);
156  cmsGetProfileInfoASCII(profile, cmsInfoDescription, "en", cmsNoCountry, profile_name, profile_name_len);
157  dest->set_profile(profile_name, data_copy, jp2_image->icc_profile_len);
158  free(profile_name);
159  } else
160  dest->set_profile("JP2 ICC profile", data_copy, jp2_image->icc_profile_len);
161  } else
162  profile = ImageFile::default_profile(cmsType);
163 
164  cmsHPROFILE lab = cmsCreateLab4Profile(NULL);
165  cmsHTRANSFORM transform = cmsCreateTransform(profile, cmsType,
166  lab, IMAGE_TYPE,
167  INTENT_PERCEPTUAL, 0);
168  cmsCloseProfile(lab);
169  cmsCloseProfile(profile);
170 
171  transform_queue queue(dest, img, T_CHANNELS(cmsType), transform);
172 
173 #pragma omp parallel shared(queue)
174  {
175  int th_id = omp_get_thread_num();
176  if (th_id == 0) { // Master thread
177  std::cerr << "\tTransforming into L*a*b* using " << omp_get_num_threads() << " threads..." << std::endl;
178  unsigned int index = 0;
179  for (unsigned int y = 0; y < img->height(); y++) {
180  void *buffer = NULL;
181  if (depth == 8)
182  buffer = planar_to_packed<unsigned char>(img->width(), T_CHANNELS(cmsType), jp2_image, index);
183  else
184  buffer = planar_to_packed<unsigned short>(img->width(), T_CHANNELS(cmsType), jp2_image, index);
185 
186  std::cerr << "\r\tRead " << (y + 1) << " of " << img->height() << " rows ("
187  << queue.num_rows() << " in queue for colour transformation) ";
188 
189  queue.add_copy(y, buffer);
190 
191  while (queue.num_rows() > 100) {
192  queue.reader_process_row();
193  std::cerr << "\r\tRead " << (y + 1) << " of " << img->height() << " rows ("
194  << queue.num_rows() << " in queue for colour transformation) ";
195  }
196  }
197  queue.set_finished();
198  while (!queue.empty()) {
199  queue.reader_process_row();
200  std::cerr << "\r\tRead " << img->height() << " of " << img->height() << " rows ("
201  << queue.num_rows() << " in queue for colour transformation) ";
202  }
203  std::cerr << std::endl;
204  } else {
205  while (!(queue.empty() && queue.finished())) {
206  while (queue.empty() && !queue.finished())
207  usleep(100);
208  queue.reader_process_row();
209  }
210  }
211  }
212  cmsDeleteTransform(transform);
213 
214  std::cerr << "Done." << std::endl;
215  return img;

218  void JP2file::write(Image::ptr img, Destination::ptr dest, bool can_free) const {
219  std::cerr << "Preparing for file " << _filepath << "..." << std::endl;
220  opj_event_mgr_t event_mgr;
221  memset(&event_mgr, 0, sizeof(opj_event_mgr_t));
222  event_mgr.error_handler = error_callback;
223  event_mgr.warning_handler = warning_callback;
224  event_mgr.info_handler = info_callback;
225 
226  opj_cparameters_t parameters;
227  opj_set_default_encoder_parameters(&parameters);
228 
229  if (dest->jp2().defined()) {
230  D_JP2 d = dest->jp2();
231  if (d.numresolutions().defined()) {
232  std::cerr << "\tUsing " << d.numresolutions() << " resolutions." << std::endl;
233  parameters.numresolution = d.numresolutions();
234  }
235  if (d.prog_order().defined()) {
236  std::string po = d.prog_order().get();
237  if (boost::iequals(po, "lrcp")) {
238  std::cerr << "\tLayer-resolution-component-precinct order." << std::endl;
239  parameters.prog_order = LRCP;
240  } else if (boost::iequals(po, "rlcp")) {
241  std::cerr << "\tResolution-layer-component-precinct order." << std::endl;
242  parameters.prog_order = RLCP;
243  } else if (boost::iequals(po, "rpcl")) {
244  std::cerr << "\tResolution-precinct-component-layer order." << std::endl;
245  parameters.prog_order = RPCL;
246  } else if (boost::iequals(po, "pcrl")) {
247  std::cerr << "\tPrecinct-component-resolution-layer order." << std::endl;
248  parameters.prog_order = PCRL;
249  } else if (boost::iequals(po, "cprl")) {
250  std::cerr << "\tComponent-precinct-resolution-layer order." << std::endl;
251  parameters.prog_order = CPRL;
252  }
253  }
254  if (d.num_rates() > 0) {
255  std::cerr << "\tRate:";
256  for (int i = 0; i < d.num_rates(); i++) {
257  parameters.tcp_rates[i] = d.rate(i);
258  if (i > 0)
259  std::cerr << ",";
260  std::cerr << " " << d.rate(i);
261  }
262  std::cerr << "." << std::endl;
263  parameters.tcp_numlayers = d.num_rates();
264  }
265  if (d.tile_size().defined()) {
266  std::cerr << "\tTile size of " << d.tile_size()->first << "" << d.tile_size()->second << std::endl;
267  parameters.cp_tdx = d.tile_size()->first;
268  parameters.cp_tdy = d.tile_size()->second;
269  }
270  }
271 
272  if (parameters.tcp_numlayers == 0) {
273  parameters.tcp_rates[0] = 0;
274  parameters.tcp_numlayers++;
275  }
276  parameters.cp_disto_alloc = 1;
277  parameters.tile_size_on = 0;
278 
279  OPJ_COLOR_SPACE colour_space;
280  unsigned char channels;
281  cmsUInt32Number cmsTempType = BYTES_SH(2);
282  if (img->is_colour()) {
283  colour_space = CLRSPC_SRGB;
284  channels = 3;
285  cmsTempType |= COLORSPACE_SH(PT_RGB);
286  } else {
287  colour_space = CLRSPC_GRAY;
288  channels = 1;
289  cmsTempType |= COLORSPACE_SH(PT_GRAY);
290  }
291  cmsTempType |= CHANNELS_SH(channels);
292 
293  int depth = 8; // Default value
294  if (dest->depth().defined())
295  depth = dest->depth();
296 
297  opj_image_cmptparm_t *components = (opj_image_cmptparm_t*)malloc(channels * sizeof(opj_image_cmptparm_t));
298  memset(components, 0, channels * sizeof(opj_image_cmptparm_t));
299  for (unsigned char i = 0; i < channels; i++) {
300  components[i].dx = parameters.subsampling_dx;
301  components[i].dy = parameters.subsampling_dy;
302  components[i].w = img->width();
303  components[i].h = img->height();
304  components[i].x0 = components[i].y0 = 0;
305  components[i].prec = components[i].bpp = depth;
306  components[i].sgnd = 0;
307  }
308  opj_image_t *jp2_image = opj_image_create(channels, components, colour_space);
309 
310  if (jp2_image == NULL)
311  return;
312 
313  cmsUInt32Number intent = INTENT_PERCEPTUAL; // Default value
314  if (dest->intent().defined())
315  intent = dest->intent();
316 
317  cmsHPROFILE profile = NULL;
318  if (dest->profile()) {
319  profile = dest->profile()->profile();
320  if (dest->profile()->has_data()) {
321  std::cerr << "\tEmbedding profile \"" << dest->profile()->name().get() << " from data (" << dest->profile()->data_size() << " bytes)." << std::endl;
322  jp2_image->icc_profile_buf = (unsigned char*)dest->profile()->data();
323  jp2_image->icc_profile_len = dest->profile()->data_size();
324  }
325  }
326  if (profile == NULL)
327  profile = this->default_profile(cmsTempType);
328 
329  cmsHPROFILE lab = cmsCreateLab4Profile(NULL);
330  cmsHTRANSFORM transform = cmsCreateTransform(lab, IMAGE_TYPE,
331  profile, cmsTempType,
332  intent, 0);
333  cmsCloseProfile(lab);
334  cmsCloseProfile(profile);
335 
336  unsigned char **rows = (unsigned char**)malloc(img->height() * sizeof(unsigned char*));
337  size_t channel_size = img->width() * (depth >> 3);
338  size_t row_size = channel_size * channels;
339  for (unsigned int y = 0; y < img->height(); y++)
340  rows[y] = (unsigned char*)malloc(row_size);
341 
342  transform_queue queue(dest, img, channels, transform);
343  if (depth == 8)
344  for (unsigned int y = 0; y < img->height(); y++)
345  queue.add(y);
346  else
347  for (unsigned int y = 0; y < img->height(); y++)
348  queue.add(y, rows[y]);
349 
350 #pragma omp parallel shared(queue)
351  {
352  int th_id = omp_get_thread_num();
353  if (th_id == 0) { // Master thread
354  std::cerr << "\tTransforming image data from L*a*b* using " << omp_get_num_threads() << " threads." << std::endl;
355 
356  Ditherer ditherer(img->width(), channels);
357 
358  unsigned int index = 0;
359  for (unsigned int y = 0; y < img->height(); y++) {
360  // Process rows until the one we need becomes available, or the queue is empty
361  {
362  short unsigned int *row = queue.row(y);
363  while (!queue.empty() && (row == NULL)) {
364  queue.writer_process_row();
365  row = queue.row(y);
366  }
367 
368  // If it's still not available, something has gone wrong
369  if (row == NULL) {
370  std::cerr << "** Oh crap (y=" << y << ", num_rows=" << queue.num_rows() << " **" << std::endl;
371  exit(2);
372  }
373 
374  if (depth == 8) {
375  ditherer.dither(row, rows[y], y == img->height() - 1);
376  queue.free_row(y);
377 
378  packed_to_planar<unsigned char>(img->width(), channels, rows[y], jp2_image, index);
379  } else
380  packed_to_planar<unsigned short>(img->width(), channels, row, jp2_image, index);
381  if (can_free)
382  img->free_row(y);
383  }
384  std::cerr << "\r\tTransformed " << y + 1 << " of " << img->height() << " rows ("
385  << queue.num_rows() << " left) ";
386  }
387  std::cerr << std::endl;
388  } else { // Other thread(s) transform the image data
389  while (!queue.empty())
390  queue.writer_process_row();
391  }
392  }
393  cmsDeleteTransform(transform);
394 
395  jp2_image->x0 = parameters.image_offset_x0;
396  jp2_image->y0 = parameters.image_offset_y0;
397  jp2_image->x1 = jp2_image->x0 + (img->width() - 1) * parameters.subsampling_dx + 1;
398  jp2_image->y1 = jp2_image->y0 + (img->height() - 1) * parameters.subsampling_dy + 1;
399 
400  parameters.tcp_mct = jp2_image->numcomps == 3 ? 1 : 0;
401 
402  opj_cinfo_t* cinfo = opj_create_compress(CODEC_JP2);
403  opj_set_event_mgr((opj_common_ptr)cinfo, &event_mgr, (void*)this);
404  opj_setup_encoder(cinfo, &parameters, jp2_image);
405  opj_cio_t *cio = opj_cio_open((opj_common_ptr)cinfo, NULL, 0);
406  opj_encode(cinfo, cio, jp2_image, NULL);
407 
408  fs::ofstream ofs(_filepath, std::ios_base::out);
409  if (ofs.fail())
410  throw FileOpenError(_filepath.native());
411 
412  int size = cio_tell(cio);
413  std::cerr << "\tWriting " << size << " bytes..." << std::endl;
414  ofs.write((char*)cio->buffer, size);
415  ofs.close();
416  opj_cio_close(cio);
417  opj_destroy_compress(cinfo);
418  opj_image_destroy(jp2_image);
419 
420  std::cerr << "Done." << std::endl;
421  }
