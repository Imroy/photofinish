36  void add_rulers(multihash& vars, std::string key, rulerlist& rulers) {
37  auto vi = vars.find(key);
38  if (vi != vars.end())
39  for (auto si : vi->second) {
40  int at = si.find_first_of('@', 0);

42  double tx = boost::lexical_cast<double>(si.substr(at + 1, si.length() - at - 1));
43  rulers.push_back(std::make_pair(fx, tx));
