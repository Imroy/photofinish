136  int webp_stream_writer::write(unsigned char* data, size_t data_size) {
137  // First write the end of any chunk in the buffer
138  if (!header_at_start) {
139  unsigned int size = min(next_chunk, data_size);
140  std::cerr << "\tWriting " << size << " bytes..." << std::endl;
141  stream->write((char*)data, size);
142  data += size;
143  data_size -= size;
144  next_chunk -= size;
145  }
146  header_at_start = false;
147 
148  // Work through every chunk in the buffer
149  while ((next_chunk < data_size) && (data_size >= 8)) {
150  memcpy(fourcc, data, 4);
151  chunk_size = read_le32(data + 4);
152  next_chunk = 8 + chunk_size + (chunk_size & 0x1);
153 
154  before_chunk();
155 
156  std::cerr << "\tFound chunk: \"" << std::string((char*)fourcc, 4) << "\", " << chunk_size << " bytes." << std::endl;
157 
158  modify_chunk(data);
159 
160  // Write a whole chunk if we have it
161  if (next_chunk < data_size) {
162  unsigned int total_size = 8 + chunk_size + (chunk_size & 0x01);
163  std::cerr << "\tWriting " << total_size << " bytes." << std::endl;
164  stream->write((char*)data, total_size);
165  data += total_size;
166  data_size -= total_size;
167  next_chunk = 0;
168  after_chunk();
169  }
170  }
171  // Write the rest of the data
172  if (data_size > 0) {
173  std::cerr << "\tWriting " << data_size << " bytes..." << std::endl;
174  stream->write((char*)data, data_size);
175  next_chunk -= data_size;
176  }
177 
178  // The end of the buffer coincides with the end of a chunk
179  if (next_chunk == 0) {
180  after_chunk();
181  header_at_start = true;
182  }
183 
184  return 1;
185  }
