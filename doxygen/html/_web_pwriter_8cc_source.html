49  void WebPwriter::write(Image::ptr img, Destination::ptr dest, bool can_free) {
50  if (_is_open)
51  throw FileOpenError("already open");
52  _is_open = true;
53 
54  std::cerr << "Preparing for file " << _filepath << "..." << std::endl;
55  std::cerr << "\t" << img->width() << "×" << img->height() << std::endl;
56 
57  WebPConfig config;
58  int ok;
59  if (dest->webp().defined()) {
60  D_WebP d = dest->webp();
61  bool need_init = true;
62 
63  if (d.preset().defined())
64  for (auto i : WebP_presets)
65  if (boost::iequals(i.first, d.preset().get())) {
66  std::cerr << "\tUsing preset \"" << d.preset().get() << "\", quality=" << d.quality() << "." << std::endl;
67  ok = WebPConfigPreset(&config, i.second, d.quality());
68  if (!ok)
69  throw WebPError(ok);
70  need_init = false;
71  break;
72  }
73  if (need_init) {
74  ok = WebPConfigInit(&config);
75  if (!ok)
76  throw WebPError(ok);
77  config.quality = d.quality();
78  std::cerr << "\tQuality=" << config.quality << std::endl;
79  }
80 
81  if (d.lossless().defined()) {
82  config.lossless = d.lossless();
83  std::cerr << "\tLossless=" << config.lossless << std::endl;
84  }
85 
86  if (d.method().defined()) {
87  config.method = d.method();
88  std::cerr << "\tMethod=" << config.method << std::endl;
89  }

92  if (!ok)
93  throw WebPError(ok);
94  }

101  if (!ok)
102  throw WebPError(ok);
103

105  pic.height = img->height();
106  pic.use_argb = config.lossless;
107  ok = WebPPictureAlloc(&pic);
108  if (!ok)
109  throw WebPError(ok);
110 
111  // libwebp takes its image data in one big chunk
112  {
113  std::cerr << "\tCopying image data to blob..." << std::endl;
114  unsigned char *rgb = new unsigned char[img->row_size() * img->height()];
115  for (unsigned int y = 0; y < img->height(); y++) {
116  memcpy(rgb + (y * img->row_size()), img->row(y), img->row_size());
117  if (can_free)

119  }
120 
121  if (img->format().extra_channels() > 0) {
122  std::cerr << "\tImporting RGBA image data..." << std::endl;
123  WebPPictureImportRGBA(&pic, rgb, img->row_size());
124  } else {

126  WebPPictureImportRGB(&pic, rgb, img->row_size());
127  }
128  delete [] rgb;
129  }

139  wrt.add_xmp(img->XMPtags());

143 
144  std::cerr << "\tEncoding..." << std::endl;
145  ok = WebPEncode(&config, &pic);
146  std::cerr << std::endl;
147  WebPPictureFree(&pic);
148
