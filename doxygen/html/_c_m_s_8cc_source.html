80  Profile::ptr Profile::sGrey(void) {
81  cmsCIExyY D65;
82  cmsWhitePointFromTemp(&D65, 6504);
83  double Parameters[5] = {
84  2.4, // Gamma
85  1.0 / 1.055, // a
86  0.055 / 1.055, // b
87  1.0 / 12.92, // c

90  // y = (x >= d ? (a*x + b)^Gamma : c*x)
91  cmsToneCurve *gamma = cmsBuildParametricToneCurve(NULL, 4, Parameters);
92  Profile::ptr profile = std::make_shared<Profile>(cmsCreateGrayProfile(&D65, gamma));
93  cmsFreeToneCurve(gamma);
94 
95  profile->write_tag(cmsSigProfileDescriptionTag, "en", "AU", "sGrey built-in");
96 
97  return profile;

374  Format &Format::set_colour_model(const ColourModel cm, unsigned int channels) {
375  _format &= COLORSPACE_MASK;
376  _format |= COLORSPACE_SH((int)cm);
377 
378  // TODO: What should be done if the supplied number of channels is wrong?
379  switch (cm) {
380  case ColourModel::Greyscale:
381  channels = 1;
382  break;
383 
384  case ColourModel::RGB:
385  case ColourModel::CMY:
386  case ColourModel::YCbCr:
387  case ColourModel::YUV:
388  case ColourModel::XYZ:
389  case ColourModel::Lab:
390  case ColourModel::HSV:
391  case ColourModel::HLS:
392  case ColourModel::Yxy:
393  case ColourModel::LabV2:
394  channels = 3;
395  break;
396 
397  case ColourModel::CMYK:
398  case ColourModel::YUVK:
399  channels = 4;
400  break;
401 
402  default:
403  // Handle the MCH* colour models
404  if ((int)cm > 14)
405  channels = (int)cm - 14;
406  break;
407  }
408 
409  if ((channels > 0) && (channels < cmsMAXCHANNELS)) {
410  _format &= CHANNELS_MASK;
411  _format |= CHANNELS_SH(channels);
412  }
413 
414  return *this;
415  }
