80  Profile::ptr Profile::sGrey(void) {
81  cmsCIExyY D65;
82  cmsWhitePointFromTemp(&D65, 6504);
83  double Parameters[5] = {
84  2.4, // Gamma
85  1.0 / 1.055, // a
86  0.055 / 1.055, // b
87  1.0 / 12.92, // c
88  0.04045, // d
89  };
90  // y = (x >= d ? (a*x + b)^Gamma : c*x)
91  cmsToneCurve *gamma = cmsBuildParametricToneCurve(NULL, 4, Parameters);
92  Profile::ptr profile = std::make_shared<Profile>(cmsCreateGrayProfile(&D65, gamma));
93  cmsFreeToneCurve(gamma);
94 
95  profile->write_MLU(cmsSigProfileDescriptionTag, "en", "AU", "sGrey built-in");
96  profile->write_MLU(cmsSigCopyrightTag, "en", "AU", "No copyright, use freely");
97 
98  return profile;
99  }

443  Format &Format::set_colour_model(const ColourModel cm, unsigned int channels) {
444  _format &= COLORSPACE_MASK;
445  _format |= COLORSPACE_SH((int)cm);
446 
447  // TODO: What should be done if the supplied number of channels is wrong?
448  switch (cm) {
449  case ColourModel::Greyscale:
450  channels = 1;
451  break;
452 
453  case ColourModel::RGB:
454  case ColourModel::CMY:
455  case ColourModel::YCbCr:
456  case ColourModel::YUV:
457  case ColourModel::XYZ:
458  case ColourModel::Lab:
459  case ColourModel::HSV:
460  case ColourModel::HLS:
461  case ColourModel::Yxy:
462  case ColourModel::LabV2:
463  channels = 3;
464  break;
465 
466  case ColourModel::CMYK:
467  case ColourModel::YUVK:
468  channels = 4;
469  break;
470 
471  default:
472  // Handle the MCH* colour models
473  if ((int)cm > 14)
474  channels = (int)cm - 14;
475  break;
476  }
477 
478  if ((channels > 0) && (channels < cmsMAXCHANNELS)) {
479  _format &= CHANNELS_MASK;
480  _format |= CHANNELS_SH(channels);
481  }
482 
483  return *this;
484  }
