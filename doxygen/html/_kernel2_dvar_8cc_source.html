73  // Template method that does the actual convolving
74  template <typename T, int channels>
75  void Kernel2Dvar::convolve_type_channels(Image::ptr src, Image::ptr dest, bool can_free) {
76 #pragma omp parallel
77  {
78 #pragma omp master
79  {
80  std::cerr << "Convolving image " << src->width() << "×" << src->height() << " => "
81  << std::setprecision(2) << std::fixed << dest->width() << "×" << dest->height()
82  << " using " << omp_get_num_threads() << " threads..." << std::endl;
83  }
84  }
85 
86  double radiusx, radiusy;
87  double norm_factx, norm_facty;
88  if (_scalex < 1.0) {
89  radiusx = this->radius();
90  norm_factx = 1.0;
91  } else {
92  radiusx = this->radius() * _scalex;
93  norm_factx = this->radius() / ceil(radiusx);
94  }
95  if (_scaley < 1.0) {
96  radiusy = this->radius();
97  norm_facty = 1.0;
98  } else {
99  radiusy = this->radius() * _scaley;
100  norm_facty = this->radius() / ceil(radiusy);
101  }
102  double aspect = radiusx / radiusy;
103 
104 #pragma omp parallel for schedule(dynamic, 1)
105  for (unsigned int ny = 0; ny < dest->height(); ny++) {
106  dest->check_rowdata_alloc(ny);
107 
108  double cy = _starty + (ny * _scaley);
109  unsigned int top = floor(cy - radiusy);
110  if (radiusy > cy)
111  top = 0;
112  unsigned int bottom = ceil(cy + radiusy);
113  if (bottom >= _from_height)
114  bottom = _from_height - 1;
115 
116  // std::cerr << "ny=" << ny << "; cy=" << cy << ", top=" << top << ", bottom=" << bottom << std::endl;
117 
118  SAMPLE temp[channels];
119  T *out = dest->row<T>(ny);
120  for (unsigned int nx = 0; nx < dest->width(); nx++) {
121  for (unsigned char c = 0; c < channels; c++)
122  temp[c] = 0;
123  SAMPLE weight = 0;
124 
125  double cx = _startx + (nx * _scalex);
126  // std::cerr << "\tnx=" << nx << "; cx=" << cx << std::endl;
127 
128  for (unsigned int y = top; y <= bottom; y++) {
129  double dy = (y - cy) * norm_facty;
130  double dy2 = sqr(dy);
131 
132  // (dx/rx)^2 + (dy/ry)^2 = 1
133  // (dx/rx)^2 = 1 - (dy/ry)^2
134  // dx^2 = rx^2 - (dy*rx/ry)^2
135  unsigned int left, right;
136  {
137  double temp = dy * aspect;
138  double w2 = sqr(radiusx) - sqr(temp);
139  if (w2 < 0)
140  continue;
141  double w = sqrt(w2);
142  left = floor(cx - w);
143  if (radiusx > cx)
144  left = 0;
145  right = ceil(cx + w);
146  if (right >= _from_width)
147  right = _from_width - 1;
148 
149  // std::cerr << "\t\ty=" << y << "; dy=" << dy << ", dy2=" << dy2 << ", w=" << w << ", left=" << left << ", right=" << right << std::endl;
150  }
151 
152  const T *in = src->at<T>(left, y);
153  for (unsigned int x = left; x <= right; x++) {
154  double dx = (x - cx) * norm_factx;
155  double dx2 = sqr(dx);
156  double r = sqrt(dx2 + dy2);
157  SAMPLE val = this->eval(r);
158  weight += val;
159  // std::cerr << "\t\t\tx=" << x << "; dx=" << dx << ", dx2=" << dx2 << ", r=" << r << ", val=" << val << ", weight=" << weight << std::endl;
160  for (unsigned char c = 0; c < channels; c++, in++)
161  temp[c] += (*in) * val;
162  }
163  }
164  // std::cerr << "\t\tweight=" << weight << std::endl;
165 
166  if (fabs(weight) > 1e-5) {
167  SAMPLE recip = 1.0 / weight;
168  for (unsigned char c = 0; c < channels; c++)
169  temp[c] *= recip;
170  }
171  for (unsigned char c = 0; c < channels; c++, out++)
172  *out = limitval<T>(temp[c]);
173  }
174 
175  if (omp_get_thread_num() == 0)
176  std::cerr << "\r\tConvolved " << ny + 1 << " of " << dest->height() << " rows";
177  }
178  std::cerr << "\r\tConvolved " << dest->height() << " of " << dest->height() << " rows." << std::endl;
179 
180  if (can_free)
181  for (unsigned int y = 0; y < _from_height; y++)
182  src->free_row(y);
183  }

185  // Template method that handles each type for convolving
186  template <typename T>
187  void Kernel2Dvar::convolve_type(Image::ptr src, Image::ptr dest, bool can_free) {
188  unsigned char channels = src->format().total_channels();
189  switch (channels) {
190  case 1: // e.g greyscale
191  convolve_type_channels<T, 1>(src, dest, can_free);
192  break;
193 
194  case 2: // e.g greyscale with alpha
195  convolve_type_channels<T, 2>(src, dest, can_free);
196  break;
197 
198  case 3: // e.g RGB, Lab, etc
199  convolve_type_channels<T, 3>(src, dest, can_free);
200  break;
201 
202  case 4: // e.g CMYK, or RGB, Lab, etc with alpha
203  convolve_type_channels<T, 4>(src, dest, can_free);
204  break;
205 
206  case 5: // e.g CMYK with alpha
207  convolve_type_channels<T, 5>(src, dest, can_free);
208  break;
209 
210  case 6:
211  convolve_type_channels<T, 6>(src, dest, can_free);
212  break;
213 
214  case 7:
215  convolve_type_channels<T, 7>(src, dest, can_free);
216  break;
217 
218  default:
219  std::cerr << "** Cannot handle " << (int)channels << " channels **" << std::endl;
220  }
221  }

224  Image::ptr Kernel2Dvar::convolve(Image::ptr img, bool can_free) {
225  auto ni = std::make_shared<Image>(_to_width_i, _to_height_i, img->format());
226  ni->set_profile(img->profile());
227 
228  if (img->xres().defined())
229  ni->set_xres(img->xres() / _scalex);
230  if (img->yres().defined())
231  ni->set_yres(img->yres() / _scaley);
232 
233  switch (img->format().bytes_per_channel()) {
234  case 1:
235  convolve_type<unsigned char>(img, ni, can_free);
236  break;
237 
238  case 2:
239  convolve_type<short unsigned int>(img, ni, can_free);
240  break;
241 
242  case 4:
243  if (img->format().is_fp())
244  convolve_type<float>(img, ni, can_free);
245  else
246  convolve_type<unsigned int>(img, ni, can_free);
247  break;
248 
249  case 8:
250  convolve_type<double>(img, ni, can_free);
251  break;
252 
253  }
254 
255  return ni;
256  }

276  SAMPLE Lanczos2D::eval(double r) const throw(Uninitialised) {
277  if (!_radius.defined())
278  throw Uninitialised("Lanczos2D", "resize.radius");
279 
280  if (fabs(r) < 1e-6)
281  return 1.0;
282  double pir = M_PI * r;
283  return (_radius * sin(pir) * sin(pir * _r_radius)) / (sqr(M_PI) * sqr(r));
284  }
